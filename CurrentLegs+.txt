        private static bool TryGetVRHand(out Transform hand)
        {
            hand = null;
            if (_vrRightHand != null && _vrRightHand.gameObject.activeInHierarchy)
            {
                hand = _vrRightHand;
                return true;
            }
            if (Time.time < _nextHandSearchTime) return false;
            _nextHandSearchTime = Time.time + HandSearchInterval;

            try
            {
                string exactPath = "FVRGlobal(Clone)/RealRig/FloorOffset/Pelvis/RightShoulder/RightArm/RightHand";
                GameObject handObj = GameObject.Find(exactPath);
                if (handObj != null)
                {
                    _vrRightHand = handObj.transform;
                    hand = handObj.transform;
                    return true;
                }

                PhysicsRigStateSync rig = UnityEngine.Object.FindObjectOfType<PhysicsRigStateSync>();
                if (rig != null)
                {
                    string[] candidates = { "PhysicsRightHand", "RightHand", "Hand_R", "RightHandAnchor" };
                    foreach (var n in candidates)
                    {
                        Transform t = rig.transform.Find(n);
                        if (t != null)
                        {
                            _vrRightHand = t;
                            hand = t;
                            return true;
                        }
                    }
                }

                FVRGrabber grabber = UnityEngine.Object.FindObjectOfType<FVRGrabber>();
                if (grabber != null && grabber.gameObject.activeInHierarchy)
                {
                    _vrRightHand = grabber.transform;
                    hand = grabber.transform;
                    return true;
                }
            }
            catch { }
            return false;
        }

        private static bool GetVRTriggerPressed()
        {
            try
            {
                UnityEngine.XR.InputDevice rightHand = UnityEngine.XR.InputDevices.GetDeviceAtXRNode(UnityEngine.XR.XRNode.RightHand);
                if (rightHand.isValid)
                {
                    if (rightHand.TryGetFeatureValue(UnityEngine.XR.CommonUsages.triggerButton, out bool pressed) && pressed) return true;
                    if (rightHand.TryGetFeatureValue(UnityEngine.XR.CommonUsages.trigger, out float val) && val > 0.35f) return true;
                }
                UnityEngine.XR.InputDevice leftHand = UnityEngine.XR.InputDevices.GetDeviceAtXRNode(UnityEngine.XR.XRNode.LeftHand);
                if (leftHand.isValid)
                {
                    if (leftHand.TryGetFeatureValue(UnityEngine.XR.CommonUsages.triggerButton, out bool pressed) && pressed) return true;
                    if (leftHand.TryGetFeatureValue(UnityEngine.XR.CommonUsages.trigger, out float val) && val > 0.35f) return true;
                }
            }
            catch { }
            return false;
        }

        // Performance fix: Cache Camera.main (internally does FindObjectOfType each call)
        private static Camera GetCachedCamera()
        {
            if (_cachedCamera == null || !_cachedCamera.isActiveAndEnabled || Time.time >= _nextCameraCacheTime)
            {
                _cachedCamera = Camera.main;
                _nextCameraCacheTime = Time.time + 1f;
            }
            return _cachedCamera;
        }

        // Performance fix: find (and cache) the local player GameObject without scanning
        // the whole scene every call. Resets the cache if the object went inactive.
        internal static GameObject FindPlayerObject()
        {
            if (_cachedPlayerObject != null && _cachedPlayerObject.activeInHierarchy)
                return _cachedPlayerObject;
            _cachedPlayerObject = null;
            try
            {
                var allGameObjects = UnityEngine.Object.FindObjectsOfType<GameObject>();
                foreach (var go in allGameObjects)
                {
                    if (go != null && go.name == "TagGameHexaPawnAudioLipSync(Clone)" && go.transform.position.sqrMagnitude > 0.01f)
                    {
                        _cachedPlayerObject = go;
                        break;
                    }
                }
            }
            catch { }
            return _cachedPlayerObject;
        }

        public override void OnUpdate()
        {
            // Freeze: zero out player Rigidbody and disable CharacterController every frame
            if (_isFrozen)
            {
                try
                {
                    GameObject player = FindPlayerObject();
                    if (player != null)
                    {
                        // Cache components only once per freeze session (or if player object changed)
                        if (_frozenCachedPlayerGO != player || _frozenCachedBehaviours == null)
                        {
                            _frozenCachedPlayerGO = player;
                            _frozenCachedCC = player.GetComponentInChildren<CharacterController>(true);
                            _frozenCachedRigidbodies = player.GetComponentsInChildren<Rigidbody>(true);
                            _frozenCachedBehaviours = new List<UnityEngine.Behaviour>();
                            foreach (var comp in player.GetComponentsInChildren<Component>(true))
                            {
                                if (comp == null) continue;
                                string compName = comp.GetType().Name;
                                if (compName.Contains("Locomotion") || compName.Contains("Movement") ||
                                    compName.Contains("Teleport") || compName.Contains("Dash"))
                                {
                                    var behaviour = comp as UnityEngine.Behaviour;
                                    if (behaviour != null) _frozenCachedBehaviours.Add(behaviour);
                                }
                            }
                        }

                        if (_frozenCachedCC != null && _frozenCachedCC.enabled) _frozenCachedCC.enabled = false;

                        if (_frozenCachedRigidbodies != null)
                        {
                            foreach (var rb in _frozenCachedRigidbodies)
                            {
                                if (rb != null)
                                {
                                    rb.velocity = Vector3.zero;
                                    rb.angularVelocity = Vector3.zero;
                                    rb.isKinematic = true;
                                }
                            }
                        }

                        foreach (var behaviour in _frozenCachedBehaviours)
                        {
                            if (behaviour != null && behaviour.enabled) behaviour.enabled = false;
                        }
                    }
                }
                catch { }
            }
            else if (_wasFrozenLastFrame)
            {
                // Unfreeze: re-enable CharacterController and non-kinematic Rigidbody
                try
                {
                    GameObject player = _frozenCachedPlayerGO ?? FindPlayerObject();
                    if (player != null)
                    {
                        if (_frozenCachedCC != null && !_frozenCachedCC.enabled) _frozenCachedCC.enabled = true;

                        if (_frozenCachedRigidbodies != null)
                        {
                            foreach (var rb in _frozenCachedRigidbodies)
                            {
                                if (rb != null) rb.isKinematic = false;
                            }
                        }

                        if (_frozenCachedBehaviours != null)
                        {
                            foreach (var behaviour in _frozenCachedBehaviours)
                            {
                                if (behaviour != null && !behaviour.enabled) behaviour.enabled = true;
                            }
                        }
                    }
                    // Clear cache
                    _frozenCachedBehaviours = null;
                    _frozenCachedRigidbodies = null;
                    _frozenCachedCC = null;
                    _frozenCachedPlayerGO = null;
                }
                catch { }
            }
            _wasFrozenLastFrame = _isFrozen;


            // Use cached panel reference
            if (_cachedMainPanel == null || !_cachedMainPanel.activeInHierarchy)
            {
                if (_isDraggingKeyboard || _isResizingKeyboard)
                {
                    _isDraggingKeyboard = false;
                    _isResizingKeyboard = false;
                    if (_keyboardPanelRt != null) _keyboardTargetPos = (Vector2)_keyboardPanelRt.localPosition;
                }
                // Allow chat keyboard to work even when GameMainPanel is inactive
                if (_keyboardMode == 4 && _keyboardContainer != null && _keyboardContainer.activeSelf && _keyboardPanelRt != null)
                {
                    // Fall through to keyboard handling below
                }
                else
                {
                    return;
                }
            }

            if (_keyboardContainer == null || !_keyboardContainer.activeSelf || _keyboardPanelRt == null)
                return;

            bool triggerNow = GetVRTriggerPressed();
            bool triggerEdge = triggerNow && !_vrTriggerWasPressed;
            _vrTriggerWasPressed = triggerNow;

            if (triggerEdge && Time.time > _dragStartTime + 0.2f)
            {
                if (_isDraggingKeyboard || _isResizingKeyboard)
                {
                    _isDraggingKeyboard = false;
                    _isResizingKeyboard = false;
                    if (_keyboardPanelRt != null) _keyboardTargetPos = (Vector2)_keyboardPanelRt.localPosition;
                }
            }

            // Proximity-based drag/resize initiation (works on any canvas, including world-space chat canvas)
            if (!_isDraggingKeyboard && !_isResizingKeyboard && triggerEdge && _keyboardPanelRt != null)
            {
                Camera initCam = GetCachedCamera();
                Transform initHand;
                if (initCam != null && TryGetVRHand(out initHand))
                {
                    Vector3 handPos = initHand.position;
                    Vector3 titlePos = _keyboardTitleBar != null
                        ? _keyboardTitleBar.transform.position
                        : _keyboardPanelRt.TransformPoint(new Vector3(0, _keyboardPanelRt.sizeDelta.y * 0.5f, 0));
                    float titleDist = Vector3.Distance(handPos, titlePos);

                    if (titleDist < 0.15f)
                    {
                        _isDraggingKeyboard = true;
                        _dragStartTime = Time.time;
                        _dragOffsetCaptured = false;
                    }
                    else
                    {
                        Vector3 resizePos = _resizeHandle != null
                            ? _resizeHandle.transform.position
                            : _keyboardPanelRt.TransformPoint(new Vector3(_keyboardPanelRt.sizeDelta.x * 0.5f, -_keyboardPanelRt.sizeDelta.y * 0.5f, 0));
                        float resizeDist = Vector3.Distance(handPos, resizePos);

                        if (resizeDist < 0.15f)
                        {
                            _isResizingKeyboard = true;
                            _keyboardResizeStartScale = _keyboardPanelRt.localScale;
                            _resizeStartDistance = Vector3.Distance(handPos, _keyboardPanelRt.position);
                            if (_resizeStartDistance < 0.001f) _resizeStartDistance = 1f;
                        }
                    }
                }
            }

            if (_isDraggingKeyboard && _keyboardPanelRt != null)
            {
                Camera cam = GetCachedCamera();
                Transform hand;
                if (cam != null && TryGetVRHand(out hand))
                {
                    Plane dragPlane = new Plane(-cam.transform.forward, _keyboardPanelRt.position);
                    Ray ray = new Ray(hand.position, hand.forward);
                    if (dragPlane.Raycast(ray, out float enter))
                    {
                        Vector3 worldHit = ray.GetPoint(enter);
                        if (_keyboardPanelRt.parent != null)
                        {
                            Vector3 localHit = _keyboardPanelRt.parent.InverseTransformPoint(worldHit);
                            localHit.z = 0f;
                            if (!_dragOffsetCaptured)
                            {
                                _keyboardDragOffset = (Vector2)localHit - (Vector2)_keyboardPanelRt.localPosition;
                                _dragOffsetCaptured = true;
                            }
                            Vector3 newPos = localHit - (Vector3)_keyboardDragOffset;
                            // FIX 2: Added 'f' suffix to all numeric literals for Mathf.Clamp
                            newPos.x = Mathf.Clamp(newPos.x, -600f, 1800f);
                            newPos.y = Mathf.Clamp(newPos.y, -900f, 500f);
                            newPos.z = 0f;
                            _keyboardPanelRt.localPosition = newPos;
                        }
                    }
                }
            }

            if (_isResizingKeyboard && _keyboardPanelRt != null)
            {
                Camera cam = GetCachedCamera();
                Transform hand;
                if (cam != null && TryGetVRHand(out hand))
                {
                    Plane dragPlane = new Plane(-cam.transform.forward, _keyboardPanelRt.position);
                    Ray ray = new Ray(hand.position, hand.forward);
                    if (dragPlane.Raycast(ray, out float enter))
                    {
                        Vector3 worldHit = ray.GetPoint(enter);
                        float currentDist = Vector3.Distance(worldHit, _keyboardPanelRt.position);
                        float startDist = _resizeStartDistance;
                        if (startDist > 0.001f)
                        {
                            float baseScale = _keyboardResizeStartScale.x;
                            float scaleFactor = baseScale * (currentDist / startDist);
                            scaleFactor = Mathf.Clamp(scaleFactor, 0.4f, 2.5f);
                            _keyboardPanelRt.localScale = new Vector3(scaleFactor, scaleFactor, 1f);
                        }
                    }
                }
            }

            if (_keyboardTitleBarImg != null && _isDraggingKeyboard != _lastTitleDragState)
            {
                _keyboardTitleBarImg.color = _isDraggingKeyboard ? _titleDragColor : _titleIdleColor;
                _lastTitleDragState = _isDraggingKeyboard;
            }
            if (_resizeHandleImg != null && _isResizingKeyboard != _lastResizeState)
            {
                _resizeHandleImg.color = _isResizingKeyboard ? _resizeActiveColor : _resizeIdleColor;
                _lastResizeState = _isResizingKeyboard;
            }
        }

        private static IEnumerator DelayedInject()
        {
            yield return new WaitForSeconds(2f);
            if (!_injected) Inject();
        }

        private static void Inject()
        {
            if (_injected) return;

            // Cache main panel references
            _cachedMainPanel = GameObject.Find("GameMainPanel(Clone)");
            if (_cachedMainPanel == null)
            {
                MelonCoroutines.Start(WaitForPanel());
                return;
            }

            _cachedTabs = _cachedMainPanel.transform.Find("Tabs");
            _cachedSubPanels = _cachedMainPanel.transform.Find("SubPanels");

            if (_cachedTabs == null || _cachedSubPanels == null)
            {
                MelonCoroutines.Start(WaitForPanel());
                return;
            }

            try
            {
                if (_cachedTabs.Find("AccountTabBtn") != null)
                {
                    _injected = true;
                    return;
                }

                _subsContainer = _cachedSubPanels;
                Transform originalTabTemplate = _cachedTabs.Find("Binds");
                if (originalTabTemplate == null)
                {
                    MelonLogger.Warning("Binds tab not found as template");
                    return;
                }

                // Create Account Tab
                GameObject tabBtn = UnityEngine.Object.Instantiate(originalTabTemplate.gameObject, _cachedTabs);
                tabBtn.name = "AccountTabBtn";
                SetText(tabBtn, "Account", 28, new Vector2(60, 30), Color.black);
                RemoveIcon(tabBtn);
                RectTransform rt = tabBtn.GetComponent<RectTransform>();
                rt.localPosition = new Vector3(570, 0, 0);

                GameObject origPanel = _cachedSubPanels.Find("BindsPanel").gameObject;
                _accountPanel = UnityEngine.Object.Instantiate(origPanel, _cachedSubPanels);
                _accountPanel.name = "AccountPanel";
                _accountPanel.transform.localScale = new Vector3(1.2f, 1.2f, 1.2f);
                for (int i = _accountPanel.transform.childCount - 1; i >= 0; i--)
                    UnityEngine.Object.DestroyImmediate(_accountPanel.transform.GetChild(i).gameObject);

                var btn = tabBtn.GetComponent<Button>();
                btn.onClick.RemoveAllListeners();
                btn.onClick.AddListener(new Action(() => {
                    for (int i = 0; i < _subsContainer.childCount; i++)
                        _subsContainer.GetChild(i).gameObject.SetActive(false);
                    _accountPanel.SetActive(true);
                    ShowAccountMenu();
                }));

                _accountContainer = new GameObject("AccountContainer");
                _accountContainer.transform.SetParent(_accountPanel.transform, false);

                // Setup Keyboard Container
                _keyboardContainer = new GameObject("KeyboardContainer");
                _keyboardContainer.AddComponent<RectTransform>();
                _keyboardContainer.transform.SetParent(_accountPanel.transform, false);
                _keyboardContainer.SetActive(false);

                CreateAccountMenu();
                _accountPanel.SetActive(false);

                InjectPersonalDataPanelToggle();
                InjectSettingsPagination();
                ModifyRoomServerPanel();
                InjectFriendsTab(_cachedTabs, _cachedSubPanels);
                InjectChatOverlay();

                // Performance fix: prewarm expensive lookups so the first list/chat build
                // doesn't run Resources.FindObjectsOfTypeAll repeatedly.
                PrewarmCaches();

                _injected = true;
                MelonLogger.Msg("Legs+ Account tab injected!");

                // Monitor tab state to hide our panels when native panels are active (LM4VR method)
                MelonCoroutines.Start(MonitorTabState());

                if (!_refresherStarted)
                {
                    _refresherStarted = true;
                    MelonCoroutines.Start(AccountRefresherLoop());
                }
            }
            catch (Exception e)
            {
                MelonLogger.Error("Injection Error: " + e.Message);
            }
        }

        // Monitor tab state - LM4VR method: hide our panels if a native panel is active
        private static IEnumerator MonitorTabState()
        {
            while (_injected)
            {
                try
                {
                bool needCheck = (_accountPanel != null && _accountPanel.activeSelf) ||
                                 (_friendsPanel != null && _friendsPanel.activeSelf);

                if (needCheck)
                {
                    if (_cachedSubPanels == null)
                    {
                        GameObject mainPanel = _cachedMainPanel ?? GameObject.Find("GameMainPanel(Clone)");
                        if (mainPanel != null)
                        {
                            _cachedSubPanels = mainPanel.transform.Find("SubPanels");
                        }
                    }
                    if (_cachedSubPanels != null)
                    {
                        for (int i = 0; i < _cachedSubPanels.childCount; i++)
                        {
                            GameObject child = _cachedSubPanels.GetChild(i).gameObject;
                            if (!child.activeSelf) continue;

                            if (_accountPanel != null && child.name != "AccountPanel" && _accountPanel.activeSelf)
                            {
                                _accountPanel.SetActive(false);
                            }
                            if (_friendsPanel != null && child.name != "FriendsPanel" && _friendsPanel.activeSelf)
                            {
                                _friendsPanel.SetActive(false);
                            }
                        }
                    }
                }
                }
                catch { }
                yield return new WaitForSeconds(0.5f);
            }
        }

        private static IEnumerator WaitForPanel()
        {
            while (!_injected)
            {
                yield return new WaitForSeconds(0.5f);
                _cachedMainPanel = GameObject.Find("GameMainPanel(Clone)");
                if (_cachedMainPanel != null) Inject();
            }
        }

        // Builds the "Status" stepper inside the Account tab (moved here from PersonalDataPanel).
        // Left/right arrows cycle online -> appear offline -> joinable -> DND and push to OnlineSystem.
        // A status dot changes colour: green=online, grey=offline, cyan=joinable, red=dnd.
        private static void CreateAccountStatusStepper(float centerX, float y)
        {
            if (_accountContainer == null) return;

            GameObject row = new GameObject("AccountStatusStepper");
            row.transform.SetParent(_accountContainer.transform, false);
            var rowRt = row.AddComponent<RectTransform>();
            rowRt.localPosition = new Vector3(centerX, y, 0f);
            rowRt.sizeDelta = new Vector2(500f, 40f);

            TextMeshProUGUI label = CreateMenuLabel("Status: Online", new Vector2(300f, 36f), new Vector3(0, 0, 0), row.transform, 24f, TextAlignmentOptions.Center);
            label.name = "StatusLabel";

            string[] statusOptions = { "online", "offline", "joinable", "dnd" };
            string[] statusLabels = { "Online", "Appear Offline", "Joinable", "DND" };
            int statusIndex = 0;

            // Start from the currently-known account status if available.
            string current = OnlineSystem.IsInitialized ? OnlineSystem.AccountStatus : "";
            for (int i = 0; i < statusOptions.Length; i++)
            {
                if (string.Equals(statusOptions[i], current, System.StringComparison.OrdinalIgnoreCase))
                {
                    statusIndex = i;
                    break;
                }
            }
            label.text = "Status: " + statusLabels[statusIndex];
            UpdateStatusDot(statusIndex);

            Action apply = () =>
            {
                label.text = "Status: " + statusLabels[statusIndex];
                UpdateStatusDot(statusIndex);
                OnlineSystem.SetAccountStatus(statusOptions[statusIndex]);
            };

            CreatePillButton("\u25C0", new Vector2(44, 36), new Vector3(-130f, 0, 0f), row.transform, () => {
                statusIndex = (statusIndex - 1 + statusOptions.Length) % statusOptions.Length;
                apply();
            });
            CreatePillButton("\u25B6", new Vector2(44, 36), new Vector3(130f, 0, 0f), row.transform, () => {
                statusIndex = (statusIndex + 1) % statusOptions.Length;
                apply();
            });
        }

        // Colours the Account-tab status dot based on the selected status index.
        private static void UpdateStatusDot(int statusIndex)
        {
            if (_accountStatusDot == null) return;
            switch (statusIndex)
            {
                case 0: _accountStatusDot.color = new Color(0.2f, 0.85f, 0.3f, 1f); break; // online = green
                case 1: _accountStatusDot.color = new Color(0.65f, 0.65f, 0.65f, 1f); break; // offline = grey
                case 2: _accountStatusDot.color = new Color(0f, 0.8f, 0.9f, 1f); break; // joinable = cyan
                case 3: _accountStatusDot.color = new Color(1f, 0.25f, 0.25f, 1f); break; // dnd = red
                default: _accountStatusDot.color = new Color(0.2f, 0.85f, 0.3f, 1f); break;
            }
        }

        private static IEnumerator AccountRefresherLoop()
        {
            while (true)
            {
                if (_accountPanel != null && _accountPanel.activeSelf)
                {
                    string newName = GetSafePlayerName();
                    if (_socialMyNameText != null && newName != _socialMyNameText.text)
                        _socialMyNameText.text = newName;

                    string newBio = OnlineSystem.IsInitialized ? OnlineSystem.Bio : _myBio;
                    if (_socialMyBioText != null && newBio != _socialMyBioText.text)
                        _socialMyBioText.text = newBio;

                    if (_socialMyRoleText != null)
                    {
                        string roleDisplay = OnlineSystem.IsInitialized ? OnlineSystem.Role.ToUpper() : "";
                        if (roleDisplay != _socialMyRoleText.text)
                        {
                            _socialMyRoleText.text = roleDisplay;
                            _socialMyRoleText.color = OnlineSystem.IsOwner ? _roleOwnerColor
                                : OnlineSystem.IsModerator ? _roleModColor : _roleDefaultColor;
                        }
                    }

                    if (_socialMyIdText != null)
                    {
                        string idStr = OnlineSystem.IsInitialized && !string.IsNullOrEmpty(OnlineSystem.LegsId) ? OnlineSystem.LegsId : GetLegsPlusIDString();
                        string newId = "ID: " + idStr;
                        if (newId != _socialMyIdText.text)
                            _socialMyIdText.text = newId;
                    }

                    // Keep the Account-tab status dot in sync with the current status.
                    string curStatus = OnlineSystem.IsInitialized ? OnlineSystem.AccountStatus : "";
                    int dotIdx = 0;
                    string[] statusOptionsSync = { "online", "offline", "joinable", "dnd" };
                    for (int i = 0; i < statusOptionsSync.Length; i++)
                    {
                        if (string.Equals(statusOptionsSync[i], curStatus, System.StringComparison.OrdinalIgnoreCase))
                        {
                            dotIdx = i;
                            break;
                        }
                    }
                    UpdateStatusDot(dotIdx);

                    string currentAvatar = GetCurrentAvatarName();
                    if (currentAvatar != _lastDetectedAvatarName)
                    {
                        _lastDetectedAvatarName = currentAvatar;
                        if (OnlineSystem.IsInitialized)
                            OnlineSystem.UpdateAvatar(currentAvatar);
                        var mirrorObj = _accountContainer?.transform.Find("ProfileFrame/InnerBackground/AvatarMirrorImage");
                        if (mirrorObj != null) MirrorAvatarToProfile(mirrorObj.gameObject);
                    }

                    // Periodically re-scan avatar names and push to Firebase
                    if (Time.realtimeSinceStartup >= _nextAvatarPushTime)
                    {
                        _nextAvatarPushTime = Time.realtimeSinceStartup + AvatarPushInterval;
                        try
                        {
                            var freshNames = GetAvailableAvatarNames();
                            if (freshNames.Count > 0 && !ListsEqual(freshNames, _avatarNames))
                            {
                                _avatarNames = freshNames;
                                string avatarJson = "[" + string.Join(",", _avatarNames.Select(n => $"\"{OnlineSystem.EscapeJson(n)}\"")) + "]";
                                OnlineSystem.FirebasePut("config/avatars", $"{{\"list\":{avatarJson}}}");
                                MelonLogger.Msg($"[Online] Re-pushed {_avatarNames.Count} avatar names to Firebase");
                            }
                            else if (freshNames.Count > 0 && _avatarNames.Count == 0)
                            {
                                _avatarNames = freshNames;
                                string avatarJson = "[" + string.Join(",", _avatarNames.Select(n => $"\"{OnlineSystem.EscapeJson(n)}\"")) + "]";
                                OnlineSystem.FirebasePut("config/avatars", $"{{\"list\":{avatarJson}}}");
                                MelonLogger.Msg($"[Online] Pushed {_avatarNames.Count} avatar names to Firebase (was empty)");
                            }
                        }
                        catch (Exception ex) { MelonLogger.Warning("[Online] Avatar re-push failed: " + ex.Message); }
                    }
                }
                yield return new WaitForSeconds(1f);
            }
        }

        internal static void ApplyAvatarInGame(string avatarName)
        {
            try
            {
                if (string.IsNullOrEmpty(avatarName)) return;
                var panelComp = GetAvatarSubPanelComponent();
                if (panelComp == null) return;
                var allToggles = panelComp.GetComponentsInChildren<Toggle>(true);
                foreach (var toggle in allToggles)
                {
                    if (toggle == null) continue;
                    string toggleName = toggle.gameObject.name.ToLower();
                    if (toggleName.Contains(avatarName.ToLower()))
                    {
                        toggle.isOn = true;
                        MelonLogger.Msg("[Online] Avatar toggle set to: " + avatarName);
                        return;
                    }
                }
                MelonLogger.Warning("[Online] Avatar toggle not found for: " + avatarName);
            }
            catch (Exception e)
            {
                MelonLogger.Warning("[Online] Could not apply avatar: " + e.Message);
            }
        }

        private static List<string> GetAvailableAvatarNames()
        {
            var names = new List<string>();
            try
            {
                var panelComp = GetAvatarSubPanelComponent();
                if (panelComp == null) return names;
                var allToggles = panelComp.GetComponentsInChildren<Toggle>(true);
                foreach (var toggle in allToggles)
                {
                    if (toggle == null) continue;
                    string toggleName = toggle.gameObject.name;
                    if (!string.IsNullOrEmpty(toggleName) && !names.Contains(toggleName))
                        names.Add(toggleName);
                }
            }
            catch { }
            return names;
        }

        private static string GetCurrentAvatarDisplayName()
        {
            if (_avatarNames.Count == 0) return "No avatars found";
            if (_avatarIndex < 0 || _avatarIndex >= _avatarNames.Count) _avatarIndex = 0;
            return _avatarNames[_avatarIndex];
        }

        private static void CycleAvatar(int direction)
        {
            if (_avatarNames.Count == 0)
            {
                _avatarNames = GetAvailableAvatarNames();
                if (_avatarNames.Count == 0) return;
            }
            _avatarIndex = (_avatarIndex + direction + _avatarNames.Count) % _avatarNames.Count;
            string avatarName = _avatarNames[_avatarIndex];
            if (_avatarLabel != null) _avatarLabel.text = avatarName;
            ApplyAvatarInGame(avatarName);
            if (OnlineSystem.IsInitialized) OnlineSystem.UpdateAvatar(avatarName);
            MelonLogger.Msg("[Account] Avatar changed to: " + avatarName);
        }

        {
            if (_nativeCircleSprite != null) return _nativeCircleSprite;
            try
            {
                var allSprites = Resources.FindObjectsOfTypeAll<Sprite>();
                foreach (var s in allSprites)
                {
                    if (s == null) continue;
                    string name = s.name.ToLower();
                    if (name == "knob" || name == "circle" || name.Contains("circle") || name.Contains("round"))
                    {
                        _nativeCircleSprite = s;
                        return s;
                    }
                }
            }
            catch { }
            return null;
        }

        private static TMP_FontAsset GetFont()
        {
            if (_cachedFont != null) return _cachedFont;
            try
            {
                var allTMP = Resources.FindObjectsOfTypeAll<TextMeshProUGUI>();
                foreach (var tmp in allTMP)
                {
                    if (tmp != null && tmp.font != null)
                    {
                        _cachedFont = tmp.font;
                        return _cachedFont;
                    }
                }
                var fontAsset = TMP_Settings.defaultFontAsset;
                if (fontAsset != null)
                {
                    _cachedFont = fontAsset;
                    return _cachedFont;
                }
            }
            catch { }
            return null;
        }

        private static Vector3 ConstrainToPanel(Vector3 position, RectTransform panel, Vector2 buttonSize)
        {
            if (panel == null) return position;
            
            float panelWidth = panel.sizeDelta.x;
            float panelHeight = panel.sizeDelta.y;
            
            float minX = -panelWidth / 2f + buttonSize.x / 2f;
            float maxX = panelWidth / 2f - buttonSize.x / 2f;
            position.x = Mathf.Clamp(position.x, minX, maxX);
            
            float minY = -panelHeight / 2f + buttonSize.y / 2f;
            float maxY = panelHeight / 2f - buttonSize.y / 2f;
            position.y = Mathf.Clamp(position.y, minY, maxY);
            
            return position;
        }

        private static bool FindVRButtonTemplate()
        {
            if (_vrButtonTemplate != null) return true;
            try
            {
                GameObject globalRoot = GameObject.Find("FVRGlobal(Clone)");
                if (globalRoot == null) return false;
                Transform t = globalRoot.transform.Find("FVRUIManager/Canvas/GameMainPanel(Clone)/SubPanels/VoiceSettingPanel/AndroidAudio/Reset Voice");
                if (t == null) return false;
                _vrButtonTemplate = t.gameObject;
                _vrButtonTemplateRect = t.GetComponent<RectTransform>();
                return true;
            }
            catch { return false; }
        }

        private static GameObject CreateVRButton(string text, float textSize, Vector2 size, Vector3 pos, Transform parent, Action onClick)
        {
            if (!FindVRButtonTemplate() || _vrButtonTemplate == null)
            {
                return CreatePillButton(text, size, pos, parent, onClick ?? (() => { }));
            }

            GameObject btnObj = UnityEngine.Object.Instantiate(_vrButtonTemplate, parent);
            btnObj.name = text + " Button";
            RectTransform rt = btnObj.GetComponent<RectTransform>();
            if (rt != null && _vrButtonTemplateRect != null)
            {
                rt.anchorMin = _vrButtonTemplateRect.anchorMin;
                rt.anchorMax = _vrButtonTemplateRect.anchorMax;
                rt.pivot = _vrButtonTemplateRect.pivot;
                rt.anchoredPosition = _vrButtonTemplateRect.anchoredPosition;
                rt.sizeDelta = _vrButtonTemplateRect.sizeDelta;
                rt.localScale = _vrButtonTemplateRect.localScale;
                rt.localRotation = _vrButtonTemplateRect.localRotation;
            }
            if (rt != null)
            {
                rt.localPosition = pos;
                rt.sizeDelta = size;
            }

            RemoveIcon(btnObj);
            SetText(btnObj, text, textSize, size, Color.white);

            Button btn = btnObj.GetComponent<Button>();
            if (btn != null)
            {
                btn.navigation = new Navigation { mode = Navigation.Mode.None };
                btn.onClick.RemoveAllListeners();
                if (onClick != null) btn.onClick.AddListener(onClick);
            }
            return btnObj;
        }

        private static void OpenKeyboard(int mode)
        {
            if (_keyboardContainer == null) return;
            _keyboardMode = mode;

            if (mode == 2)
            {
                if (_friendsPanel != null && _keyboardContainer.transform.parent != _friendsPanel.transform)
                    _keyboardContainer.transform.SetParent(_friendsPanel.transform, false);
            }
            else if (mode == 3)
            {
                if (_subsContainer != null)
                {
                    Transform roomPanel = _subsContainer.Find("RoomServerPanel");
                    if (roomPanel != null && _keyboardContainer.transform.parent != roomPanel)
                        _keyboardContainer.transform.SetParent(roomPanel, false);
                }
            }
            else if (mode == 4)
            {
                // Chat keyboard lives on the chat overlay panel
                if (_chatPanel != null && _keyboardContainer.transform.parent != _chatPanel.transform)
                    _keyboardContainer.transform.SetParent(_chatPanel.transform, false);
                PauseChatAutoHideTimer();
            }
            else if (mode == 5)
            {
                // Friends search keyboard lives on the friends panel
                if (_friendsPanel != null && _keyboardContainer.transform.parent != _friendsPanel.transform)
                    _keyboardContainer.transform.SetParent(_friendsPanel.transform, false);
            }
            else
            {
                if (_accountPanel != null && _keyboardContainer.transform.parent != _accountPanel.transform)
                    _keyboardContainer.transform.SetParent(_accountPanel.transform, false);
            }

            if (mode == 0) _currentInput = GetSafePlayerName();
            else if (mode == 1) _currentInput = _myBio;
            else if (mode == 2) _currentInput = _dmInputText;
            else if (mode == 4) _currentInput = _chatInputText;
            else _currentInput = "";

            _cursorIndex = _currentInput.Length;
            _isUppercase = true;
            CreateKeyboardUI();
            _keyboardContainer.SetActive(true);
        }

        private static void InsertNameText(string text)
        {
            if (string.IsNullOrEmpty(text)) return;
            if (_keyboardMode == 0)
            {
                char c = text[0];
                if (!char.IsLetterOrDigit(c) && c != '-' && c != '_' && c != '.')
                    return;
            }
            if (_keyboardMode == 3)
            {
                char c = text[0];
                if (!char.IsLetterOrDigit(c) && c != '-')
                    return;
            }
            if (_keyboardMode != 3)
            {
                int limit = _keyboardMode == 2 ? DmMessageCharLimit : (_keyboardMode == 1 ? 100 : (_keyboardMode == 4 ? ChatMessageCharLimit : 32));
                if (_currentInput.Length >= limit) return;
            }
            _currentInput = _currentInput.Insert(_cursorIndex, text);
            _cursorIndex += text.Length;
            UpdateInputDisplay();
        }

        private static void MoveCursor(int direction)
        {
            _cursorIndex += direction;
            if (_cursorIndex < 0) _cursorIndex = 0;
            if (_cursorIndex > _currentInput.Length) _cursorIndex = _currentInput.Length;
            UpdateInputDisplay();
        }

        private static void UpdateInputDisplay()
        {
            if (_cursorIndex > _currentInput.Length) _cursorIndex = _currentInput.Length;
            if (_cursorIndex < 0) _cursorIndex = 0;
            string visual = _currentInput.Insert(_cursorIndex, "|");
            string prefix = _keyboardMode == 0 ? "Name: "
                          : _keyboardMode == 1 ? "Bio: "
                          : _keyboardMode == 2 ? "Message: "
                          : _keyboardMode == 4 ? "Chat: "
                          : _keyboardMode == 5 ? "Search: "
                          : "Input ServerID: ";
            if (_inputDisplay != null)
            {
                TextMeshProUGUI tmp = _inputDisplay.GetComponent<TextMeshProUGUI>();
                if (tmp != null) tmp.text = prefix + visual;
            }
            if (_nameCounterText != null)
            {
                int limit = _keyboardMode == 2 ? DmMessageCharLimit : (_keyboardMode == 1 ? 100 : (_keyboardMode == 4 ? ChatMessageCharLimit : 32));
                _nameCounterText.text = $"{_currentInput.Length} / {limit}";
                _nameCounterText.color = _currentInput.Length >= limit ? Color.red : Color.black;
            }

            // Update command mode and player list panel visibility
            if (_keyboardMode == 4)
            {
                if (_currentInput.StartsWith("/mt")) _chatCommandMode = 3;
                else if (_currentInput.StartsWith("/t")) _chatCommandMode = 2;
                else if (_currentInput.StartsWith("/mh")) _chatCommandMode = 4;
                else if (_currentInput.StartsWith("/m")) _chatCommandMode = 1;
                else _chatCommandMode = 0;
                UpdateChatPlayerListPanel();
            }
        }

        {
            string cleaned = SanitizePlayerName(_currentInput);
            if (_keyboardMode == 0)

                MelonLogger.Msg("Bio updated to: " + cleaned);
                if (_socialMyBioText != null) _socialMyBioText.text = cleaned;

            }

        private static string SanitizePlayerName(string value)
        {
            if (string.IsNullOrEmpty(value)) return "";
            var sb = new System.Text.StringBuilder();
            int limit = _keyboardMode == 2 ? DmMessageCharLimit : (_keyboardMode == 1 ? 100 : (_keyboardMode == 4 ? ChatMessageCharLimit : 32));
            for (int i = 0; i < value.Length && sb.Length < limit; i++)
            {
                char c = value[i];
                if (char.IsWhiteSpace(c))
                {
                    if (_keyboardMode == 1 || _keyboardMode == 2 || sb.Length > 0) sb.Append(c);
                }
                else if (c != '<' && c != '>' && c != '=' && c != '*' && !char.IsControl(c))
                {
                    sb.Append(c);
                }
            }
            return sb.ToString().TrimEnd();
        }

        private static string StripRichText(string input)
        {
            if (string.IsNullOrEmpty(input)) return "";
            var sb = new System.Text.StringBuilder(input.Length);
            bool inside = false;
            for (int i = 0; i < input.Length; i++)
            {
                char c = input[i];
                if (c == '<') inside = true;
                else if (c == '>') inside = false;
                else if (!inside) sb.Append(c);
            }
            return sb.ToString();
        }

        private static string GetCurrentAvatarName()
        {
            if (Time.time < _nextAvatarDetectTime && !string.IsNullOrEmpty(_cachedAvatarName))
                return _cachedAvatarName;

            string result = null;

            try
            {
                var info = DetectActiveAvatarInfo();
                if (info != null && !string.IsNullOrEmpty(info.AvatarName))
                    result = info.AvatarName.Trim();
            }
            catch { }

            if (string.IsNullOrEmpty(result))
            {
                try
                {
                    result = DetectActiveAvatarFromUI();
                }
                catch { }
            }

            if (string.IsNullOrEmpty(result))
            {
                try
                {
                    result = DetectAvatarFromSubPanel();
                }
                catch { }
            }

            if (!string.IsNullOrEmpty(result))
            {
                _cachedAvatarName = result;
                _nextAvatarDetectTime = Time.time + AvatarDetectInterval;
                return result;
            }

            return string.IsNullOrEmpty(_cachedAvatarName) ? "Shade" : _cachedAvatarName;
        }

        private static string DetectAvatarFromSubPanel()
        {
            try
            {
                var panelComp = GetAvatarSubPanelComponent();
                if (panelComp == null) return null;
                
                // Find the currently selected toggle in the avatar list
                var toggles = panelComp.GetComponentsInChildren<Toggle>(true);
                if (toggles == null) return null;
                foreach (var toggle in toggles)
                {
                    if (toggle != null && toggle.isOn)
                    {
                        string name = toggle.gameObject.name;
                        if (!string.IsNullOrEmpty(name) && name != "Toggle")
                        {
                            // Clean up the name - remove "Avatar_" prefix if present
                            if (name.StartsWith("Avatar_")) name = name.Substring(7);
                            return name.Trim();
                        }
                    }
                }
                
                // Fallback: check for any active avatar container
                var contentRoot = panelComp.transform.Find("AvatarScrollView/Viewport/Content");
                if (contentRoot != null)
                {
                    for (int i = 0; i < contentRoot.childCount; i++)
                    {
                        Transform child = contentRoot.GetChild(i);
                        if (child == null) continue;
                        var tgl = child.GetComponentInChildren<Toggle>(true);
                        if (tgl != null && tgl.isOn)
                        {
                            string name = child.name;
                            if (name.StartsWith("Avatar_")) name = name.Substring(7);
                            return name.Trim();
                        }
                    }
                }
            }
            catch { }
            return null;
        }

        private static string DetectActiveAvatarFromUI()
        {
            try
            {
                var panelComp = GetAvatarSubPanelComponent();
                if (panelComp == null) return null;
                var toggles = panelComp.GetComponentsInChildren<Toggle>(true);
                if (toggles == null) return null;
                foreach (var toggle in toggles)
                {
                    if (toggle != null && toggle.isOn)
                    {
                        string name = toggle.gameObject.name.ToLower().Trim();
                        if (name.Contains("avatar") || name.Contains("shade") || name.Length > 2)
                        {
                            return toggle.gameObject.name;
                        }
                    }
                }
            }
            catch { }
            return null;
        }

        private static Il2CppXRIF._Core.Rigs.AvatarInfo DetectActiveAvatarInfo()
        {
            try
            {
                // Cache player object for performance
                GameObject player = FindPlayerObject();
                if (player == null) return null;

                var components = player.GetComponentsInChildren<Component>(true);
                if (components == null) return null;

                foreach (var comp in components)
                {
                    if (comp == null) continue;
                    var compType = comp.GetIl2CppType();
                    var fields = compType.GetFields(Il2CppSystem.Reflection.BindingFlags.Instance | Il2CppSystem.Reflection.BindingFlags.Public | Il2CppSystem.Reflection.BindingFlags.NonPublic);
                    foreach (var f in fields)
                    {
                        if (f.FieldType.Name.Contains("AvatarInfo"))
                        {
                            var val = f.GetValue(comp);
                            if (val != null)
                            {
                                var avatarInfo = val.TryCast<Il2CppXRIF._Core.Rigs.AvatarInfo>();
                                if (avatarInfo != null) return avatarInfo;
                            }
                        }
                    }
                    var props = compType.GetProperties(Il2CppSystem.Reflection.BindingFlags.Instance | Il2CppSystem.Reflection.BindingFlags.Public | Il2CppSystem.Reflection.BindingFlags.NonPublic);
                    foreach (var p in props)
                    {
                        if (p.PropertyType.Name.Contains("AvatarInfo") && p.CanRead)
                        {
                            var val = p.GetValue(comp);
                            if (val != null)
                            {
                                var avatarInfo = val.TryCast<Il2CppXRIF._Core.Rigs.AvatarInfo>();
                                if (avatarInfo != null) return avatarInfo;
                            }
                        }
                    }
                }
            }
            catch { }
            return null;
        }

        private static void MirrorAvatarToProfile(GameObject targetRawImageObj)
        {
            try
            {
                var targetRawImage = targetRawImageObj.GetComponent<RawImage>();
                if (targetRawImage == null) return;

                var activeInfo = DetectActiveAvatarInfo();
                if (activeInfo != null)
                {
                    if (activeInfo.HeadIcon != null && activeInfo.HeadIcon.texture != null)
                    {
                        targetRawImage.texture = activeInfo.HeadIcon.texture;
                        targetRawImage.color = Color.white;
                        MelonLogger.Msg($"[Legs+] Profile card updated using native head icon for: '{activeInfo.AvatarName}'");
                        return;
                    }
                }

                string activeAvatarName = GetCurrentAvatarName().ToLower().Trim();
                if (_cachedMainPanel == null) return;

                Transform contentRoot = _cachedMainPanel.transform.Find("SubPanels/AvatarSubPanel/AvatarScrollView/Viewport/Content");
                if (contentRoot == null) return;

                Texture2D clonedTexture = null;
                Sprite clonedSprite = null;

                for (int i = 0; i < contentRoot.childCount; i++)
                {
                    Transform child = contentRoot.GetChild(i);
                    if (child == null) continue;
                    string itemName = child.name.ToLower().Trim();
                    bool isSelectedCard = false;
                    var toggle = child.GetComponentInChildren<Toggle>(true);
                    if (toggle != null && toggle.isOn) isSelectedCard = true;

                    if (itemName.Contains(activeAvatarName) || isSelectedCard)
                    {
                        var rawImage = child.GetComponentInChildren<RawImage>(true);
                        if (rawImage != null && rawImage.texture != null)
                        {
                            clonedTexture = rawImage.texture.TryCast<Texture2D>();
                            break;
                        }
                        var image = child.GetComponentInChildren<Image>(true);
                        if (image != null && image.sprite != null)
                        {
                            clonedSprite = image.sprite;
                            break;
                        }
                    }
                }

                if (clonedTexture != null)
                {
                    targetRawImage.texture = clonedTexture;
                    targetRawImage.color = Color.white;
                }
                else if (clonedSprite != null)
                {
                    targetRawImage.texture = clonedSprite.texture;
                    targetRawImage.color = Color.white;
                }
                else
                {
                    targetRawImage.color = new Color(0.25f, 0.25f, 0.25f, 1f);
                }
            }
            catch (Exception ex)
            {
                MelonLogger.Warning("[MirrorAvatar] Error: " + ex.Message);
            }
        }

        private static bool ListsEqual(List<string> a, List<string> b)
        {
            if (a.Count != b.Count) return false;
            for (int i = 0; i < a.Count; i++)
                if (!string.Equals(a[i], b[i], System.StringComparison.OrdinalIgnoreCase))
                    return false;
            return true;
        }

        // ----------------------------------------------------------------
        // ESP & Rainbow Pulse
        // ----------------------------------------------------------------

        // Helper: clones the "Enable Rolling" toggle template, positions it below the
        // previous row, hides its checkmark, and wires onValueChanged.
        // Returns the new RectTransform so callers can chain extra setup.
        //
        // TO ADD A NEW SETTING TOGGLE: just call this once, e.g.
        //   AddPersonalDataToggle(toggleTemplate, "MyToggle", "My Label", prevRt, origRt, (on) => {
        //       // do something when toggled
        //   });
        // It auto-handles cloning, positioning, checkmark visuals and event wiring.
        private static RectTransform AddPersonalDataToggle(
            Transform template, string name, string label, RectTransform prevRt, RectTransform origRt,
            Action<bool> onChanged)
        {
            GameObject toggleObj = UnityEngine.Object.Instantiate(template.gameObject, template.parent);
            toggleObj.name = name;
            var labelTxt = toggleObj.GetComponentInChildren<TextMeshProUGUI>();
            if (labelTxt != null) labelTxt.text = label;

            RectTransform rt = toggleObj.GetComponent<RectTransform>();
            if (rt != null && origRt != null)
            {
                rt.localPosition = (prevRt != null ? prevRt.localPosition : origRt.localPosition)
                    + new Vector3(0, -(origRt.sizeDelta.y + 5f), 0);
            }

            Il2CppArrayBase<Image> images = toggleObj.GetComponentsInChildren<Image>(true);
            foreach (var img in images)
            {
                if (img != null && (img.name == "Checkmark" || img.name.ToLower().Contains("check")))
                    img.gameObject.SetActive(false);
            }

            Toggle toggle = toggleObj.GetComponentInChildren<Toggle>(true);
            if (toggle != null)
            {
                Image bg = toggleObj.transform.Find("Background")?.GetComponent<Image>();
                if (bg != null) bg.raycastTarget = true;
                toggle.navigation = new Navigation { mode = Navigation.Mode.None };
                toggle.transition = Selectable.Transition.None;
                toggle.interactable = true;
                toggle.onValueChanged.RemoveAllListeners();
                toggle.SetIsOnWithoutNotify(false);
                toggle.onValueChanged.AddListener(new Action<bool>(onChanged));
            }

            return rt;
        }

            }
        }

        private static System.Collections.IEnumerator ClearRecentlySentTextAfterDelay(string text, float delay)
        {
            yield return new WaitForSeconds(delay);

        private static IEnumerator ChatAutoHideLoop()
        {
            while (true)
            {
                if (_chatAutoHideActive && !_chatAutoHidePaused && _chatShowMode == 0)
                {
                    _chatAutoHideTimer -= Time.deltaTime;
                    if (_chatAutoHideTimer <= 0f)
                    {
                        yield return ChatFadeOut();
                        _chatAutoHideActive = false;
                    }
                }
                yield return null;
            }
        }

        private static IEnumerator ChatFadeOut()
        {
            if (_chatOverlay == null) yield break;
            CanvasGroup cg = _chatOverlay.GetComponent<CanvasGroup>();
            if (cg == null) yield break;
            float duration = 0.5f;
            float elapsed = 0f;
            float startAlpha = cg.alpha;
            while (elapsed < duration)
            {
                elapsed += Time.deltaTime;
                cg.alpha = Mathf.Lerp(startAlpha, 0f, elapsed / duration);
                yield return null;
            }
            cg.alpha = 0f;
            _chatOverlay.SetActive(false);
        }

        private static void SendChatMessage(string sender, string text)
        {
            if (string.IsNullOrEmpty(text)) return;

            string myName = GetSafePlayerName();

            // Check for slash commands
            if (text.StartsWith("/"))
            {
                string cmdPart = text.Substring(1);
                int spaceIdx = cmdPart.IndexOf(' ');
                string cmd = (spaceIdx >= 0 ? cmdPart.Substring(0, spaceIdx) : cmdPart).ToLower();
                string msgText = spaceIdx >= 0 ? cmdPart.Substring(spaceIdx + 1).Trim() : "";

                switch (cmd)
                {
                    case "m":
                        // Message to all players - bubble shows [User] : Message
                        string mNetMsg = "[" + myName + "] : " + msgText;
                        LM4VR_SendToAllPlayers(mNetMsg);
                        _recentlySentTexts.Add(mNetMsg);
                        MelonCoroutines.Start(ClearRecentlySentTextAfterDelay(mNetMsg, 5f));
                        _chatMessages.Add(new ChatMessage { Sender = myName, Text = msgText, Timestamp = DateTime.Now, Type = ChatType.ServerAll });
                        break;
                    case "mh":
                        // Message to all (hidden/special)
                        LM4VR_SendToAllPlayers(msgText);
                        _recentlySentTexts.Add(msgText);
                        MelonCoroutines.Start(ClearRecentlySentTextAfterDelay(msgText, 5f));
                        _chatMessages.Add(new ChatMessage { Sender = myName, Text = msgText, Timestamp = DateTime.Now, Type = ChatType.ModMessage });
                        break;
                    case "t":
                        // Local-only whisper (mod-only online sending for later)
                        if (_chatTargetNames.Count > 0)
                        {
                            string targetList = string.Join(", ", _chatTargetNames);
                            _chatMessages.Add(new ChatMessage { Sender = targetList, Text = msgText, Timestamp = DateTime.Now, Type = ChatType.Whisper });
                        }
                        else
                        {
                            _chatMessages.Add(new ChatMessage { Sender = "SYSTEM", Text = "No players targeted. Select players first.", Timestamp = DateTime.Now, Type = ChatType.ServerHidden });
                        }
                        break;
                    case "mt":
                        // Networked whisper to any player (regardless of mod)
                        if (_chatTargetNames.Count > 0)
                        {
                            var playerRefs = LM4VR_ScanPlayers();
                            int sentCount = 0;
                            string whisperNetMsg = "<i>" + myName + " whispered to you</i> : " + msgText;
                            foreach (var pInfo in playerRefs)
                            {
                                if (_chatTargetNames.Contains(pInfo.Name))
                                {
                                    object pRef = pInfo.PlayerRef;
                                    if (pRef == null) pRef = LM4VR_FindPlayerRefByName(pInfo.Name);
                                    if (pRef != null)
                                    {
                                        if (LM4VR_TrySendMessageToPlayerRef(pRef, whisperNetMsg, 8))
                                            sentCount++;
                                    }
                                    else
                                    {
                                        MelonLogger.Warning("[Chat] Could not find PlayerRef for: " + pInfo.Name);
                                    }
                                }
                            }
                            _recentlySentTexts.Add(whisperNetMsg);
                            MelonCoroutines.Start(ClearRecentlySentTextAfterDelay(whisperNetMsg, 5f));
                            string targetList = string.Join(", ", _chatTargetNames);
                            _chatMessages.Add(new ChatMessage { Sender = targetList, Text = msgText, Timestamp = DateTime.Now, Type = ChatType.Whisper });
                            MelonLogger.Msg("[Chat] Whisper sent to " + sentCount + "/" + _chatTargetNames.Count + " player(s)");
                        }
                        else
                        {
                            _chatMessages.Add(new ChatMessage { Sender = "SYSTEM", Text = "No players targeted. Select players first.", Timestamp = DateTime.Now, Type = ChatType.ServerHidden });
                        }
                        break;
                    default:
                        // Unknown command
                        _chatMessages.Add(new ChatMessage { Sender = "SYSTEM", Text = "Unknown command: /" + cmd, Timestamp = DateTime.Now, Type = ChatType.ServerHidden });
                        break;
                }
            }
            else
            {
                // Normal chat (local only for now)
                _chatMessages.Add(new ChatMessage { Sender = sender, Text = text, Timestamp = DateTime.Now, Type = ChatType.Normal });
            }

            RebuildChatMessages();

            // Show chat when a message arrives
            if (_chatOverlay != null && !_chatOverlay.activeSelf)
            {
                _chatOverlay.SetActive(true);
                CanvasGroup cg = _chatOverlay.GetComponent<CanvasGroup>();
                if (cg != null) cg.alpha = 0.95f;
            }

            if (_chatShowMode == 0) StartChatAutoHideTimer();
        }

        internal static void ReceiveChatMessage(string sender, string text)
        {
            ReceiveChatMessage(sender, text, ChatType.Normal);
        }

        internal static void ReceiveChatMessage(string sender, string text, ChatType type)
        {
            if (string.IsNullOrEmpty(text)) return;

            if (_recentlySentTexts.Contains(text)) return;

            for (int i = _chatMessages.Count - 1; i >= 0; i--)
            {
                var existing = _chatMessages[i];
                if (existing.Sender == sender && existing.Text == text
                    && (DateTime.Now - existing.Timestamp).TotalSeconds < 3)
                    return;
            }

            _chatMessages.Add(new ChatMessage { Sender = sender, Text = text, Timestamp = DateTime.Now, Type = type });
            RebuildChatMessages();

            if (_chatOverlay != null && _chatEnabled && _chatShowMode == 0)
            {
                _chatOverlay.SetActive(true);
                CanvasGroup cg = _chatOverlay.GetComponent<CanvasGroup>();
                if (cg != null) cg.alpha = 0.95f;
            }

            if (_chatShowMode == 0) StartChatAutoHideTimer();

            MelonLogger.Msg($"[Chat] Received from {sender}: {text}");
        }

        private static void RebuildChatMessages()
        {
            if (_chatMessagesContent == null) return;

            for (int i = _chatMessagesContent.childCount - 1; i >= 0; i--)
                UnityEngine.Object.DestroyImmediate(_chatMessagesContent.GetChild(i).gameObject);

            int msgIndex = 0;

            foreach (var msg in _chatMessages)
            {
                bool isMe = msg.Sender == "me";
                bool isServer = msg.Sender == "SERVER";
                string senderName = isMe ? GetSafePlayerName() : msg.Sender;

                GameObject msgObj = new GameObject("Msg_" + msgIndex);
                msgObj.transform.SetParent(_chatMessagesContent, false);
                var msgRt = msgObj.AddComponent<RectTransform>();
                LayoutElement msgLe = msgObj.AddComponent<LayoutElement>();
                msgLe.minHeight = 20f;

                TextMeshProUGUI mTmp = msgObj.AddComponent<TextMeshProUGUI>();
                mTmp.fontSize = 16;
                mTmp.alignment = TextAlignmentOptions.Left;
                mTmp.enableWordWrapping = true;
                if (_cachedFont != null) mTmp.font = _cachedFont;

                mTmp.color = Color.white;

                switch (msg.Type)
                {
                    case ChatType.ServerAll:
                        mTmp.text = "<color=#FFC800>[SERVER:</color><color=#2ADB50>" + senderName + "</color><color=#FFC800>]</color> : " + msg.Text;
                        break;
                    case ChatType.ServerHidden:
                        mTmp.text = "<color=#00FFFF>" + msg.Text + "</color>";
                        break;
                    case ChatType.ModMessage:
                        mTmp.text = "<color=#FFC800>[SERVER]</color> : " + msg.Text;
                        break;
                    case ChatType.Whisper:
                        if (_whisperTagEnabled)
                            mTmp.text = "<color=#FF80FF><i>[You whispered to " + senderName + "]</i></color> : " + msg.Text;
                        else
                            mTmp.text = "<color=#FF80FF>" + msg.Text + "</color>";
                        break;
                    case ChatType.WhisperAll:
                        mTmp.text = "<color=#00FFFF>[Server]</color> : " + msg.Text;
                        break;
                    case ChatType.Alert:
                        mTmp.text = msg.Text;
                        break;
                    default: // Normal
                        if (isServer)
                            mTmp.text = "<color=#FFC800>[SERVER]</color> : " + msg.Text;
                        else if (isMe)
                            mTmp.text = "<color=#2ADB50>[" + GetSafePlayerName() + "]</color> : " + msg.Text;
                        else
                            mTmp.text = "<color=#4DA6FF>[" + senderName + "]</color> : " + msg.Text;
                        break;
                }

                msgIndex++;
            }

            Canvas.ForceUpdateCanvases();
            if (_chatScrollRect != null)
                _chatScrollRect.normalizedPosition = new Vector2(0, 0);
        }

        private static object LM4VR_CreateDefaultForParameter(System.Reflection.ParameterInfo p)
        {
            try
            {
                if (p.HasDefaultValue) return p.DefaultValue;
            }
            catch { }
            try
            {
                if (p.ParameterType == typeof(int) || p.ParameterType.Name == "Int32") return 0;
                if (p.ParameterType == typeof(float) || p.ParameterType.Name == "Single") return 0f;
                if (p.ParameterType == typeof(bool) || p.ParameterType.Name == "Boolean") return false;
                if (p.ParameterType.IsEnum) return Enum.ToObject(p.ParameterType, 0);
                return Activator.CreateInstance(p.ParameterType);
            }
            catch { }
            return null;
        }

        internal static bool LM4VR_TrySendMessageToPlayerRef(object playerRef, string message, int timeout)
        {
            try
            {
                if (playerRef == null) return false;
                MessageSO messageSO = LM4VR_GetMessageSO();
                if (messageSO == null) return false;

                System.Reflection.MethodInfo sendToPlayerMethod = null;
                foreach (var m in messageSO.GetType().GetMethods())
                {
                    if (m.Name == "SendToPlayer")
                    {
                        sendToPlayerMethod = m;
                        break;
                    }
                }
                if (sendToPlayerMethod == null) return false;

                var parms = sendToPlayerMethod.GetParameters();
                var args = new object[parms.Length];
                bool usedRef = false;
                bool usedMessage = false;
                for (int i = 0; i < parms.Length; i++)
                {
                    string typeName = parms[i].ParameterType.Name.ToLower();
                    if (!usedRef && (typeName.Contains("playerref") || typeName.Contains("player")))
                    {
                        args[i] = playerRef;
                        usedRef = true;
                    }
                    else if (!usedMessage && parms[i].ParameterType == typeof(string))
                    {
                        args[i] = message;
                        usedMessage = true;
                    }
                    else if (parms[i].ParameterType == typeof(int) || parms[i].ParameterType.Name == "Int32")
                    {
                        args[i] = timeout;
                    }
                    else if (parms[i].ParameterType.IsEnum)
                    {
                        args[i] = Enum.ToObject(parms[i].ParameterType, 0);
                    }
                    else
                    {
                        args[i] = LM4VR_CreateDefaultForParameter(parms[i]);
                    }
                }
                if (parms.Length >= 2 && (!usedRef || !usedMessage))
                {
                    args[0] = playerRef;
                    args[1] = message;
                }
                sendToPlayerMethod.Invoke(messageSO, args);
                return true;
            }
            catch (Exception ex)
            {
                MelonLogger.Warning("[Chat] SendToPlayer failed: " + ex.Message);
                return false;
            }
        }

        internal static object LM4VR_FindPlayerRefByName(string playerName)
        {
            try
            {
                var identities = Resources.FindObjectsOfTypeAll<PlayerIdentity>();
                if (identities == null) return null;
                for (int i = 0; i < identities.Count; i++)
                {
                    PlayerIdentity pid = identities[i];
                    if (pid == null) continue;
                    string name = pid.PlayerName ?? "";
                    if (name != playerName) continue;
                    GameObject go = pid.gameObject;
                    if (go == null) continue;
                    NetworkObject no = go.GetComponent<NetworkObject>();
                    if (no == null) no = go.GetComponentInParent<NetworkObject>();
                    if (no == null) no = go.GetComponentInChildren<NetworkObject>(true);
                    if (no == null) continue;
                    var prop = no.GetType().GetProperty("InputAuthority");
                    if (prop == null) continue;
                    object pRef = prop.GetValue(no);
                    if (pRef != null)
                    {
                        string refStr = pRef.ToString();
                        if (!string.IsNullOrEmpty(refStr) && !refStr.Contains("None") && refStr != "0")
                            return pRef;
                    }
                }
            }
            catch { }
            return null;
        }

        private static void LM4VR_SendToPlayer(object playerRef, string text)
        {
            LM4VR_TrySendMessageToPlayerRef(playerRef, text, 8);
        }

        internal struct PlayerInfo
        {
            public string Name;
            public object PlayerRef;
        }

        internal static List<PlayerInfo> LM4VR_ScanPlayers()
        {
            var result = new List<PlayerInfo>();
            try
            {
                var identities = Resources.FindObjectsOfTypeAll<PlayerIdentity>();
                if (identities == null || identities.Count == 0) return result;

                HashSet<string> seenUIDs = new HashSet<string>();
                for (int i = 0; i < identities.Count; i++)
                {
                    try
                    {
                        PlayerIdentity playerIdentity = identities[i];
                        if (playerIdentity == null) continue;

                        string pName = playerIdentity.PlayerName ?? "Unknown";
                        string uid = playerIdentity.UnityPlayerId ?? "0";
                        if (seenUIDs.Contains(uid)) continue;
                        seenUIDs.Add(uid);

                        // Get PlayerRef via NetworkObject.InputAuthority
                        object playerRef = null;
                        try
                        {
                            GameObject go = playerIdentity.gameObject;
                            if (go != null)
                            {
                                NetworkObject no = go.GetComponent<NetworkObject>();
                                if (no == null) no = go.GetComponentInParent<NetworkObject>();
                                if (no != null)
                                {
                                    var inputAuthProp = no.GetType().GetProperty("InputAuthority");
                                    if (inputAuthProp != null)
                                    {
                                        playerRef = inputAuthProp.GetValue(no);
                                        if (playerRef != null)
                                        {
                                            string refStr = playerRef.ToString();
                                            if (string.IsNullOrEmpty(refStr) || refStr.Contains("None") || refStr == "0")
                                                playerRef = null;
                                        }
                                    }
                                }
                            }
                        }
                        catch { }

                        result.Add(new PlayerInfo { Name = pName, PlayerRef = playerRef });
                    }
                    catch { }
                }
            }
            catch { }
            return result;
        }

        private static string LM4VR_GetPlayerName(object playerRef)
        {
            var players = LM4VR_ScanPlayers();
            foreach (var p in players)
            {
                if (p.PlayerRef != null && p.PlayerRef.Equals(playerRef))
                    return p.Name;
            }
            return "Unknown";
        }

        private static bool _bubbleHookStarted = false;
        private static readonly Dictionary<int, string> _seenBubbleTexts = new();
        private static FVRMessageBubbleManager _cachedBubbleManager;
        private static FVRMessageBubble[] _cachedBubbleInstances;
        private static Transform _cachedMessageRoot;
        private static float _nextBubbleSearchTime;
        private const float BubbleSearchInterval = 2f;

        private static void StartBubblePolling()
        {
            if (_bubbleHookStarted) return;
            _bubbleHookStarted = true;
            MelonCoroutines.Start(BubbleMessagePollLoop());
        }

        private static IEnumerator BubbleMessagePollLoop()
        {
            yield return new WaitForSeconds(3f); // Initial delay

            while (true)
            {
                if (_chatEnabled && _chatShowMode != 2)
                {
                    try
                    {
                        PollForBubbleMessages();
                    }
                    catch (Exception e)
                    {
                        MelonLogger.Warning("[Chat] Bubble poll error: " + e.Message);
                    }
                }
                yield return new WaitForSeconds(0.5f); // Faster polling to catch bubble messages
            }
        }

        private static void PollForBubbleMessages()
        {
            // Strategy 1: Try to find FVRMessageBubbleManager
            if (_cachedBubbleManager == null)
            {
                try
                {
                    var managers = UnityEngine.Object.FindObjectsOfType<FVRMessageBubbleManager>();
                    if (managers != null && managers.Length > 0)
                    {
                        _cachedBubbleManager = managers[0];
                        MelonLogger.Msg("[Chat] Found FVRMessageBubbleManager");
                    }
                }
                catch { }
            }

            if (_cachedBubbleManager != null)
            {
                PollBubbleManager();
            }

            // Strategy 2: Also check FVRMessageBubble instances directly
            if (Time.time >= _nextBubbleSearchTime)
            {
                _nextBubbleSearchTime = Time.time + BubbleSearchInterval;
                try
                {
                    _cachedBubbleInstances = UnityEngine.Object.FindObjectsOfType<FVRMessageBubble>();
                }
                catch { }
            }

            if (_cachedBubbleInstances != null)
            {
                PollBubbleInstances();
            }

            // Strategy 3: Fallback - check MessageRoot path
            if (_cachedMessageRoot == null && Time.time >= _nextBubbleSearchTime)
            {
                try
                {
                    GameObject itemBag = GameObject.Find("FVRGlobal(Clone)/RealRig/FloorOffset/BagParent/ItemBag");
                    if (itemBag != null)
                    {
                        Transform msgRoot = itemBag.transform.Find("MessageRoot");
                        if (msgRoot != null)
                        {
                            _cachedMessageRoot = msgRoot;
                            MelonLogger.Msg("[Chat] Found MessageRoot");
                        }
                    }
                }
                catch { }
            }

            if (_cachedMessageRoot != null)
            {
                PollMessageRoot();
            }
        }

        private static void PollBubbleManager()
        {
            try
            {
                if (_cachedBubbleManager == null) return;
                var bubble = _cachedBubbleManager.MessageBubble;
                if (bubble == null) return;

                // Read text from _messageUi children
                if (bubble._messageUi == null) return;
                var tmps = bubble._messageUi.GetComponentsInChildren<TextMeshProUGUI>(true);
                if (tmps == null) return;

                foreach (var tmp in tmps)
                {
                    if (tmp == null || string.IsNullOrEmpty(tmp.text)) continue;
                    string msgText = tmp.text.Trim();
                    if (msgText.Length < 3 || msgText.Length > 500) continue;
                    if (tmp.transform.IsChildOf(_chatOverlay?.transform)) continue;

                    int hash = msgText.GetHashCode();
                    if (_seenBubbleTexts.ContainsKey(hash)) continue;

                    _seenBubbleTexts[hash] = msgText;
                    ReceiveChatMessage("SERVER", msgText);
                }
            }
            catch
            {
                _cachedBubbleManager = null;
            }
        }

        private static void PollBubbleInstances()
        {
            try
            {
                if (_cachedBubbleInstances == null) return;
                foreach (var bubble in _cachedBubbleInstances)
                {
                    if (bubble == null) continue;

                    // Check _messageUi
                    if (bubble._messageUi != null)
                    {
                        var tmps = bubble._messageUi.GetComponentsInChildren<TextMeshProUGUI>(true);
                        if (tmps != null)
                        {
                            foreach (var tmp in tmps)
                            {
                                if (tmp == null || string.IsNullOrEmpty(tmp.text)) continue;
                                string msgText = tmp.text.Trim();
                                if (msgText.Length < 3 || msgText.Length > 500) continue;
                                if (tmp.transform.IsChildOf(_chatOverlay?.transform)) continue;

                                int hash = msgText.GetHashCode();
                                if (_seenBubbleTexts.ContainsKey(hash)) continue;

                                _seenBubbleTexts[hash] = msgText;
                                ReceiveChatMessage("SERVER", msgText);
                            }
                        }
                    }

                    // Also check Root object for TMP text
                    if (bubble.Root != null)
                    {
                        var tmps = bubble.Root.GetComponentsInChildren<TextMeshProUGUI>(true);
                        if (tmps != null)
                        {
                            foreach (var tmp in tmps)
                            {
                                if (tmp == null || string.IsNullOrEmpty(tmp.text)) continue;
                                string msgText = tmp.text.Trim();
                                if (msgText.Length < 3 || msgText.Length > 500) continue;
                                if (tmp.transform.IsChildOf(_chatOverlay?.transform)) continue;

                                int hash = msgText.GetHashCode();
                                if (_seenBubbleTexts.ContainsKey(hash)) continue;

                                _seenBubbleTexts[hash] = msgText;
                                ReceiveChatMessage("SERVER", msgText);
                            }
                        }
                    }
                }
            }
            catch { }
        }

        private static void PollMessageRoot()
        {
            try
            {
                if (_cachedMessageRoot == null) return;
                var tmps = _cachedMessageRoot.GetComponentsInChildren<TextMeshProUGUI>(true);

        }

        private static void ESPDisable()
        {
            if (_espLensObject != null)
            {
                try { UnityEngine.Object.Destroy(_espLensObject); }
                catch { }
                _espLensObject = null;
            }
            _espLensRigidbodies = null;

        }

        private static IEnumerator ESPFollowLoop()
        {
            while (_isESPActive && _espLensObject != null)
            {
                if (_espAnchorTransform == null)
                {
                    Camera cam = GetCachedCamera();
                    _espAnchorTransform = cam != null ? cam.transform : null;
                }
                if (_espAnchorTransform != null)
                {
                    _espLensObject.transform.position = _espAnchorTransform.position + (_espAnchorTransform.forward * _espLensFaceDistance);
                    _espLensObject.transform.rotation = _espAnchorTransform.rotation;
                    ZeroESPRigidbodies();
                }
                yield return null;
            }
        }

        private static void ESPEnable()
        {
            GameObject peekLensPrefab = FindPeekLensPrefab();
            if (peekLensPrefab == null)
            {
                MelonLogger.Error("ESP: PeekLens prefab not found!");
                _isESPActive = false;
                return;
            }

            try
            {
                _espLensObject = UnityEngine.Object.Instantiate(peekLensPrefab);
                _espLensObject.name = "ESP_PeekLens_Local";
                var netObj = _espLensObject.GetComponent<NetworkObject>();
                if (netObj != null) UnityEngine.Object.DestroyImmediate(netObj);

                var allBehaviours = _espLensObject.GetComponentsInChildren<MonoBehaviour>(true);
                foreach (var c in allBehaviours)
                {
                    if (c == null) continue;
                    try
                    {
                        string typeName = c.GetIl2CppType().Name;
                        if (typeName.Contains("Network") || typeName.Contains("FVRNetwork") ||
                            typeName.Contains("FVRFusion") || typeName.Contains("RPC"))
                        {
                            UnityEngine.Object.DestroyImmediate(c);
                        }
                    }
                    catch { }
                }

                _espLensRigidbodies = _espLensObject.GetComponentsInChildren<Rigidbody>(true);
                foreach (var rb in _espLensRigidbodies)
                {
                    if (rb != null)
                    {
                        rb.isKinematic = true;
                        rb.useGravity = false;
                        rb.velocity = Vector3.zero;
                        rb.angularVelocity = Vector3.zero;
                    }
                }

                var colliders = _espLensObject.GetComponentsInChildren<Collider>(true);
                foreach (var c in colliders)
                {
                    if (c != null) c.enabled = false;
                }

                for (int i = _espLensObject.transform.childCount - 1; i >= 0; i--)
                {
                    Transform child = _espLensObject.transform.GetChild(i);
                    if (child.name == "MM_Handle")
                        UnityEngine.Object.DestroyImmediate(child.gameObject);
                    for (int j = child.childCount - 1; j >= 0; j--)
                    {
                        Transform grandchild = child.GetChild(j);
                        if (grandchild.name == "MM_Handle")
                            UnityEngine.Object.DestroyImmediate(grandchild.gameObject);
                    }
                }

                _espLensObject.SetActive(true);
                var renderers = _espLensObject.GetComponentsInChildren<Renderer>(true);
                foreach (var r in renderers)
                {
                    if (r == null) continue;
                    r.enabled = true;
                    foreach (var mat in r.materials)
                    {
                        if (mat != null) mat.renderQueue = 1;
                    }
                }

                Camera startCam = GetCachedCamera();
                _espAnchorTransform = startCam != null ? startCam.transform : null;
                if (_espAnchorTransform != null)
                {
                    _espLensObject.transform.SetParent(null, true);
                    _espLensObject.transform.position = _espAnchorTransform.position + (_espAnchorTransform.forward * _espLensFaceDistance);
                    _espLensObject.transform.rotation = _espAnchorTransform.rotation;
                }

                ZeroESPRigidbodies();
                MelonLogger.Msg("ESP: Client-side PeekLens created!");
                MelonCoroutines.Start(ESPFollowLoop());
            }
            catch (Exception ex)
            {
                MelonLogger.Error("ESP Enable error: " + ex.Message);
                _isESPActive = false;
            }
        }

        private static void ZeroESPRigidbodies()
        {
            if (_espLensRigidbodies == null) return;
            foreach (Rigidbody rb in _espLensRigidbodies)
            {
                if (rb == null) continue;
                rb.velocity = Vector3.zero;
                rb.angularVelocity = Vector3.zero;
            }
        }

        private static GameObject FindPeekLensPrefab()
        {
            Il2CppArrayBase<NetworkObject> allNetObjs = Resources.FindObjectsOfTypeAll<NetworkObject>();
            GameObject fallback = null;
            foreach (var no in allNetObjs)
            {
                if (no == null) continue;
                if (no.name != "PeekLens") continue;
                if (no.gameObject.activeInHierarchy) continue;
                if (no.GetComponent<Rigidbody>() != null)
                    return no.gameObject;
                if (fallback == null)
                    fallback = no.gameObject;
            }
            return fallback;
        }

                            dispRt.anchorMin = new Vector2(0.5f, 0.5f);
                            dispRt.anchorMax = new Vector2(0.5f, 0.5f);
                            dispRt.pivot = new Vector2(0.5f, 0.5f);
                            dispRt.localPosition = new Vector3(-175f, 155f, 0f);
                            dispRt.localScale = new Vector3(1.2f, 1.2f, 1.1f);
                        }
                    }

                    var allCloneButtons = keyboardClone.GetComponentsInChildren<Button>(true);
                    foreach (var b in allCloneButtons)
                    {
                        if (b == null) continue;
                        var lbl = b.GetComponentInChildren<TextMeshProUGUI>(true);
                        if (lbl == null) continue;
                        string label = lbl.text.Trim();
                        b.onClick.RemoveAllListeners();

                        if (label.Length == 1 && char.IsLetterOrDigit(label[0]))
                        {
                            char ch = label[0];
                            b.onClick.AddListener(new Action(() => {
                                string letter = char.ToUpper(ch).ToString();
                                _serverCurrentInput = _serverCurrentInput.Insert(_serverCursorIndex, letter);
                                _serverCursorIndex += letter.Length;
                                if (_serverKeyboardInputDisplay != null)
                                    _serverKeyboardInputDisplay.text = "Input ServerID: " + _serverCurrentInput.Insert(_serverCursorIndex, "|");
                                if (_serverIdDisplayText != null)
                                    _serverIdDisplayText.text = _serverCurrentInput;
                            }));
                        }
                        else if (label == "<-")
                        {
                            b.onClick.AddListener(new Action(() => {
                                if (_serverCursorIndex > 0 && _serverCurrentInput.Length > 0)
                                {
                                    _serverCurrentInput = _serverCurrentInput.Remove(_serverCursorIndex - 1, 1);
                                    _serverCursorIndex--;
                                    if (_serverKeyboardInputDisplay != null)
                                        _serverKeyboardInputDisplay.text = "Input ServerID: " + _serverCurrentInput.Insert(_serverCursorIndex, "|");
                                    if (_serverIdDisplayText != null)
                                        _serverIdDisplayText.text = _serverCurrentInput;
                                }
                            }));
                        }
                        else if (label == "#")
                        {
                            b.onClick.AddListener(new Action(() => {
                                _serverCurrentInput = _serverCurrentInput.Insert(_serverCursorIndex, "#");
                                _serverCursorIndex++;
                                if (_serverKeyboardInputDisplay != null)
                                    _serverKeyboardInputDisplay.text = "Input ServerID: " + _serverCurrentInput.Insert(_serverCursorIndex, "|");
                                if (_serverIdDisplayText != null)
                                    _serverIdDisplayText.text = _serverCurrentInput;
                            }));
                        }
                    }

                    _serverKeyboardContainer = keyboardClone;
                    _serverKeyboardPanelObj = keyboardClone;
                    MelonLogger.Msg("[ServerBrowser] Server keyboard clone placed with " + allCloneButtons.Length + " re-wired buttons");
                }

                var allPanelBtns = panel.GetComponentsInChildren<Button>(true);
                foreach (var b in allPanelBtns)
                {
                    if (b == null) continue;
                    var btnLabel = b.GetComponentInChildren<TextMeshProUGUI>(true);
                    string btnName = b.gameObject.name;
                    if (btnLabel != null && (btnLabel.text.Contains("Enter") || btnName.Contains("Enter")))
                    {
                        b.onClick.RemoveAllListeners();

                            string code = _serverCurrentInput;
                            if (string.IsNullOrEmpty(code)) return;
                            JoinServer(code);

                            _serverCursorIndex = 0;
                            if (_serverKeyboardInputDisplay != null)

                                _serverIdDisplayText.text = "";
                        }));

                        break;
                    }
                }

                CloseKeyboard();
            }
            catch (Exception ex)
            {
                MelonLogger.Warning("[ServerBrowser] Could not replace server keyboard: " + ex.Message);
            }

            RefreshServerList();
            MelonLogger.Msg("[ServerBrowser] RoomServerPanel ready");
        }

        private static void RefreshServerList()
        {
            _discoveredServers.Clear();
            try
            {
                // Cache DebugStartSO
                Il2CppReferenceArray<UnityEngine.Object> arr = Resources.FindObjectsOfTypeAll(Il2CppType.Of<DebugStartSO>());
                if (arr == null || arr.Count == 0) { MelonLogger.Warning("[ServerBrowser] No DebugStartSO found"); return; }
                DebugStartSO ds = new DebugStartSO(arr[0].Pointer);
                if (ds == null) { MelonLogger.Warning("[ServerBrowser] DebugStartSO is null"); return; }

                var roomList = ds.SameSceneModelRoomList;
                if (roomList == null) { MelonLogger.Warning("[ServerBrowser] SameSceneModelRoomList is null"); return; }

                for (int i = 0; i < roomList.Count; i++)
                {
                    SessionItem s = roomList[i];
                    if (s == null) continue;
                    _discoveredServers.Add(new ServerEntry
                    {
                        SessionKey = s.SessionKey,
                        SceneGameMode = s.SceneGameMode.ToString(),
                        PlayerCount = s.PlayerCount,
                        MaxPlayers = s.MaxPlayers
                    });
                }
                MelonLogger.Msg($"[ServerBrowser] Found {_discoveredServers.Count} servers");
            }
            catch (Exception ex)
            {
                MelonLogger.Error("[ServerBrowser] RefreshServerList error: " + ex.Message);
            }

            try { PopulateServerEntries(); }
            catch (Exception ex) { MelonLogger.Error("[ServerBrowser] PopulateServerEntries error: " + ex.Message); }
        }

        private static void PopulateServerEntries()
        {
            if (_friendListRoot == null)
            {
                MelonLogger.Error("[ServerBrowser] _friendListRoot is null, cannot populate");
                return;
            }

            Transform container = _friendListRoot;
            int clearedCount = 0;
            for (int i = container.childCount - 1; i >= 0; i--)
            {
                Transform child = container.GetChild(i);
                if (_friendServerEntryTemplate != null && child.gameObject == _friendServerEntryTemplate)
                    continue;
                UnityEngine.Object.DestroyImmediate(child.gameObject);
                clearedCount++;
            }

            if (clearedCount > 0)
                MelonLogger.Msg($"[ServerBrowser] Cleared {clearedCount} old entries, template kept");

            if (_friendServerEntryTemplate != null)
                _friendServerEntryTemplate.SetActive(false);

            Canvas.ForceUpdateCanvases();

            if (_discoveredServers.Count == 0)
            {
                MelonLogger.Msg("[ServerBrowser] No servers to display");
                return;
            }

            if (_friendServerEntryTemplate == null)
            {
                MelonLogger.Error("[ServerBrowser] Entry template is null, cannot create entries");
                return;
            }

            MelonLogger.Msg($"[ServerBrowser] Populating {_discoveredServers.Count} entries from '{_friendServerEntryTemplate.name}'");

            float entryHeight = 40f;
            RectTransform templateRt = _friendServerEntryTemplate.GetComponent<RectTransform>();
            if (templateRt != null && templateRt.rect.height > 1f)
                entryHeight = templateRt.rect.height;
            entryHeight = Math.Max(entryHeight, 70f);

            float startY = 0f;
            for (int i = 0; i < _discoveredServers.Count; i++)
            {
                ServerEntry session = _discoveredServers[i];
                int index = i;
                GameObject entry = UnityEngine.Object.Instantiate(_friendServerEntryTemplate, container);
                entry.name = "ServerEntry_" + index;

                if (_itemUIType != null)
                {
                    Component itemComp = entry.GetComponent(_itemUIType);
                    if (itemComp != null)
                        UnityEngine.Object.Destroy(itemComp);
                }

                RectTransform entryRt = entry.GetComponent<RectTransform>();
                if (entryRt != null)
                {
                    entryRt.anchorMin = new Vector2(0, 1);
                    entryRt.anchorMax = new Vector2(1, 1);
                    entryRt.pivot = new Vector2(0.5f, 1);
                    entryRt.localPosition = new Vector3(0, startY - (index * entryHeight), 0);
                }

                entry.SetActive(true);

                var texts = entry.GetComponentsInChildren<TextMeshProUGUI>(true);
                TextMeshProUGUI btnTxt = null;
                foreach (var t in texts)
                {
                    if (t != null && t.transform.parent != null &&
                        t.transform.parent.GetComponent<FVRButton>() != null)
                    {
                        btnTxt = t;
                        break;
                    }
                }

                bool hasSetMainText = false;
                foreach (var t in texts)
                {
                    if (t == null || t == btnTxt) continue;
                    t.color = Color.black;
                    if (!hasSetMainText)
                    {
                        t.text = $"<size=35><b>{session.SessionKey}</b></size><br>Mode: {session.SceneGameMode} | Players: {session.PlayerCount}/{session.MaxPlayers}";
                        t.fontSize = 28f;
                        t.enableWordWrapping = true;
                        hasSetMainText = true;
                    }
                    else
                    {
                        t.text = "";
                    }
                }

                if (btnTxt != null)
                {
                    btnTxt.text = "Go";
                    btnTxt.color = Color.white;
                }

                Button btn = entry.GetComponentInChildren<Button>(true);
                if (btn != null)
                {
                    btn.navigation = new Navigation { mode = Navigation.Mode.None };
                    btn.onClick.RemoveAllListeners();
                    string sessionKey = session.SessionKey;
                    btn.onClick.AddListener(new Action(() => JoinServer(sessionKey)));
                }
            }

            RectTransform contentRt = container.GetComponent<RectTransform>();
            if (contentRt != null)
                contentRt.sizeDelta = new Vector2(contentRt.sizeDelta.x, _discoveredServers.Count * entryHeight);

            Canvas.ForceUpdateCanvases();
            MelonLogger.Msg($"[ServerBrowser] {_discoveredServers.Count} entries populated");
        }

        internal static void JoinServer(string code)
        {
            if (string.IsNullOrEmpty(code) || code == "...") return;
            try
            {
                Il2CppSystem.Type serverPanelType = Il2CppSystem.Type.GetType("_Common.Gameplay.Support.Scripts.InGameConsole.ServerConsole.ServerPanel, Il2CppGameplay.Support.HU");
                if (serverPanelType == null)
                    serverPanelType = Il2CppSystem.Type.GetType("_Common.Gameplay.Support.Scripts.InGameConsole.ServerConsole.ServerPanel, Gameplay.Support.HU");
                if (serverPanelType == null) { MelonLogger.Error("ServerPanel type not found"); return; }

                var allPanels = Resources.FindObjectsOfTypeAll(serverPanelType);
                for (int i = 0; i < allPanels.Count; i++)
                {
                    var sp = allPanels[i].TryCast<Component>();
                    if (sp != null && sp.gameObject.name == "Server")
                    {
                        var enterMeth = serverPanelType.GetMethod("EnterServer");
                        if (enterMeth == null) { MelonLogger.Error("EnterServer method not found"); return; }
                        enterMeth.Invoke(sp.TryCast<Il2CppSystem.Object>(), new Il2CppSystem.Object[] { (Il2CppSystem.String)code });
                        OnlineSystem.UpdateCurrentCode(code);
                        MelonLogger.Msg($"Joining server: {code}");
                        return;
                    }
                }
                MelonLogger.Error("ServerPanel not found - cannot join");
            }
            catch (Exception ex)
            {
                MelonLogger.Error("JoinServer failed: " + ex.Message);
            }
        }

        private static void RefreshFriendsList()
        {
            if (_friendsListContent == null) return;

            // Performance fix: skip full rebuild when the underlying data is unchanged
            int hash = ComputeFriendsListHash();
            if (hash == _lastFriendsListHash)
                return;
            _lastFriendsListHash = hash;

            RebuildFriendsList();
        }

        private static int ComputeFriendsListHash()
        {
            unchecked
            {
                int h = 17;
                h = h * 31 + (_friendsSearchInput == null ? 0 : _friendsSearchInput.GetHashCode());
                foreach (var f in _friendsList)
                {
                    h = h * 31 + f.Username.GetHashCode();
                    h = h * 31 + f.LegsId.GetHashCode();
                    h = h * 31 + (int)f.Status;
                }
                h = h * 31 + _pendingRequests.Count;
                foreach (var f in _pendingRequests)
                {
                    h = h * 31 + f.Username.GetHashCode();
                    h = h * 31 + f.LegsId.GetHashCode();
                    h = h * 31 + (int)f.Status;
                }
                return h;
            }
        }

        private static void RebuildFriendsList()
        {
            for (int i = _friendsListContent.childCount - 1; i >= 0; i--)
                UnityEngine.Object.DestroyImmediate(_friendsListContent.GetChild(i).gameObject);

            if (_requestsListContent != null)
            {
                for (int i = _requestsListContent.childCount - 1; i >= 0; i--)
                    UnityEngine.Object.DestroyImmediate(_requestsListContent.GetChild(i).gameObject);
            }

            // --- Populate Friends (left) ---
            List<FriendEntry> friendsDisplay = new List<FriendEntry>(_friendsList);
            if (!string.IsNullOrEmpty(_friendsSearchInput))
            {
                string q = _friendsSearchInput.ToLower();
                friendsDisplay = _friendsList.FindAll(f =>
                    f.Username.ToLower().Contains(q) ||
                    f.LegsId.ToLower().Contains(q));
            }

            if (friendsDisplay.Count == 0)

                emptyObj.transform.SetParent(_friendsListContent, false);

                foreach (var f in friendsDisplay)

                    BuildFriendEntry(f, _friendsListContent, isPending: false);
                }
            }

            // --- Populate Requests (right) ---
            if (_requestsListContent != null)
            {
                if (_pendingRequests.Count == 0)
                {
                    GameObject emptyObj = new GameObject("EmptyText");
                    emptyObj.transform.SetParent(_requestsListContent, false);
                    emptyObj.AddComponent<RectTransform>();

                    emptyTmp.fontSize = 28f;

                    }
                }
            }

            Canvas.ForceUpdateCanvases();
        }

        internal static void SendFriendRequest(string targetLegsId, string targetUsername)
        {
            if (string.IsNullOrEmpty(targetLegsId) || string.IsNullOrEmpty(OnlineSystem.LegsId)) return;
            string myId = OnlineSystem.LegsId;
            string myName = OnlineSystem.Username;
            string data = $"{{\"username\":\"{OnlineSystem.EscapeJson(myName)}\",\"fromId\":\"{OnlineSystem.EscapeJson(myId)}\",\"timestamp\":{OnlineSystem.GetCurrentTimestamp()}}}";
            bool ok = OnlineSystem.FirebasePut($"friendRequests/{Uri.EscapeDataString(targetLegsId)}/{Uri.EscapeDataString(myId)}", data);
            if (ok) MelonLogger.Msg("[Friends] Friend request sent to " + targetUsername);
            else MelonLogger.Warning("[Friends] Failed to send friend request to " + targetUsername);
        }

        private static void AcceptFriendRequest(string fromId, string fromUsername)
        {
            if (string.IsNullOrEmpty(fromId)) return;
            string myId = OnlineSystem.LegsId;
            OnlineSystem.FirebaseDelete($"friendRequests/{Uri.EscapeDataString(myId)}/{Uri.EscapeDataString(fromId)}");
            string data1 = $"{{\"username\":\"{OnlineSystem.EscapeJson(fromUsername)}\",\"legsid\":\"{OnlineSystem.EscapeJson(fromId)}\"}}";
            string data2 = $"{{\"username\":\"{OnlineSystem.EscapeJson(OnlineSystem.Username)}\",\"legsid\":\"{OnlineSystem.EscapeJson(myId)}\"}}";
            OnlineSystem.FirebasePut($"friends/{Uri.EscapeDataString(myId)}/{Uri.EscapeDataString(fromId)}", data1);
            OnlineSystem.FirebasePut($"friends/{Uri.EscapeDataString(fromId)}/{Uri.EscapeDataString(myId)}", data2);
            _friendsList.Add(new FriendEntry { Username = fromUsername, LegsId = fromId, Status = FriendStatus.Offline });
            _pendingRequests.RemoveAll(r => r.LegsId == fromId);
            RefreshFriendsList();
            MelonLogger.Msg("[Friends] Accepted friend request from " + fromUsername);
        }

        private static void DeclineFriendRequest(string fromId)
        {
            if (string.IsNullOrEmpty(fromId)) return;
            string myId = OnlineSystem.LegsId;
            OnlineSystem.FirebaseDelete($"friendRequests/{Uri.EscapeDataString(myId)}/{Uri.EscapeDataString(fromId)}");
            _pendingRequests.RemoveAll(r => r.LegsId == fromId);
            RefreshFriendsList();
            MelonLogger.Msg("[Friends] Declined friend request from " + fromId);
        }

        internal static IEnumerator PollFriendStatuses()
        {
            yield return new WaitForSeconds(10f);
            while (true)
            {
                yield return new WaitForSeconds(15f);
                if (!OnlineSystem.IsInitialized || string.IsNullOrEmpty(OnlineSystem.LegsId)) continue;
                foreach (var f in _friendsList)
                {
                    if (string.IsNullOrEmpty(f.LegsId)) continue;
                    try
                    {
                        string json = OnlineSystem.FirebaseGet($"online/{Uri.EscapeDataString(f.LegsId)}");
                        if (!string.IsNullOrEmpty(json) && json != "null")
                        {
                            bool isOnline = json.Contains("\"isOnline\":true");
                            string currentCode = "";
                            var codeMatch = System.Text.RegularExpressions.Regex.Match(json, "\"currentCode\":\"([^\"]+)\"");
                            if (codeMatch.Success) currentCode = codeMatch.Groups[1].Value;
                            if (isOnline && !string.IsNullOrEmpty(currentCode))
                                f.Status = FriendStatus.InGame;
                            else if (isOnline)
                                f.Status = FriendStatus.Online;
                            else
                                f.Status = FriendStatus.Offline;
                            f.CurrentCode = currentCode;
                        }
                        else
                        {
                            f.Status = FriendStatus.Offline;
                        }
                    }
                    catch { f.Status = FriendStatus.Offline; }
                }
                if (_friendsPanel != null && _friendsPanel.activeSelf)
                    RefreshFriendsList();
            }
        }

        internal static void LoadPendingFriendRequests()
        {
            if (string.IsNullOrEmpty(OnlineSystem.LegsId)) return;
            try
            {
                string json = OnlineSystem.FirebaseGet($"friendRequests/{Uri.EscapeDataString(OnlineSystem.LegsId)}");
                if (string.IsNullOrEmpty(json) || json == "null") return;
                var ids = OnlineSystem.ExtractAllUserIds(json);
                foreach (var id in ids)
                {
                    if (_pendingRequests.Any(r => r.LegsId == id)) continue;
                    string userJson = OnlineSystem.FirebaseGet($"users/{Uri.EscapeDataString(id)}/username");
                    string username = !string.IsNullOrEmpty(userJson) && userJson != "null" ? userJson.Trim('"') : id;
                    _pendingRequests.Add(new FriendEntry { Username = username, LegsId = id });
                }
                MelonLogger.Msg("[Friends] Loaded " + _pendingRequests.Count + " pending friend requests");
            }
            catch (Exception e)
            {
                MelonLogger.Warning("[Friends] Failed to load pending requests: " + e.Message);
            }
        }

        internal static void LoadFriendsList()
        {
            if (string.IsNullOrEmpty(OnlineSystem.LegsId)) return;
            try
            {
                string json = OnlineSystem.FirebaseGet($"friends/{Uri.EscapeDataString(OnlineSystem.LegsId)}");
                if (string.IsNullOrEmpty(json) || json == "null") return;
                _friendsList.Clear();
                var matches = System.Text.RegularExpressions.Regex.Matches(json, @"""([^""{}]+)"":\s*\{""username"":""([^""]+)""");
                foreach (System.Text.RegularExpressions.Match m in matches)
                {
                    string id = m.Groups[1].Value;
                    string username = m.Groups[2].Value;
                    if (!_friendsList.Any(f => f.LegsId == id))
                        _friendsList.Add(new FriendEntry { Username = username, LegsId = id });
                }
                MelonLogger.Msg("[Friends] Loaded " + _friendsList.Count + " friends");
                RefreshFriendsList();
            }
            catch (Exception e)
            {
                MelonLogger.Warning("[Friends] Failed to load friends list: " + e.Message);
            }
        }

        private static void SendDMMessage(string text)
        {
            if (string.IsNullOrEmpty(_dmTargetUser)) return;
            if (!_dmThreads.TryGetValue(_dmTargetUser, out var thread))
            {
                thread = new List<DMMessage>();
                _dmThreads[_dmTargetUser] = thread;
            }

            thread.Add(new DMMessage { Sender = "me", Text = text, Timestamp = DateTime.Now });

            foreach (var f in _friendsList)
            {
                if (f.Username == _dmTargetUser)
                {
                    f.LastMessage = FormatLastMessagePreview("me", text);
                    break;
                }
            }

            RebuildDMMessages();
            RefreshFriendsList();
        }

        private static void ReceiveDMMessage(string fromUser, string text)
        {
            if (string.IsNullOrEmpty(fromUser) || string.IsNullOrEmpty(text)) return;
            if (!_dmThreads.TryGetValue(fromUser, out var thread))
            {
                thread = new List<DMMessage>();
                _dmThreads[fromUser] = thread;
            }

            thread.Add(new DMMessage { Sender = fromUser, Text = text, Timestamp = DateTime.Now });

            foreach (var f in _friendsList)
            {
                if (f.Username == fromUser)
                {
                    f.LastMessage = FormatLastMessagePreview(fromUser, text);
                    break;
                }
            }

            if (_dmTargetUser == fromUser && _dmPanel != null && _dmPanel.activeSelf)
                RebuildDMMessages();
            RefreshFriendsList();

            MelonLogger.Msg($"[DM] Received from {fromUser}: {text}");
        }

        private static void RebuildDMMessages()
        {
            if (_dmMessagesContent == null || string.IsNullOrEmpty(_dmTargetUser)) return;

            try
            {
                for (int i = _dmMessagesContent.childCount - 1; i >= 0; i--)
                    UnityEngine.Object.DestroyImmediate(_dmMessagesContent.GetChild(i).gameObject);

                string myName = GetSafePlayerName();

                if (!_dmThreads.TryGetValue(_dmTargetUser, out var thread)) return;

                string lastSender = "";
                DateTime lastTimestamp = DateTime.MinValue;
                int msgIndex = 0;
                foreach (var msg in thread)
                {
                    bool isMe = msg.Sender == "me";
                    string senderName = isMe ? myName : msg.Sender;
                    bool senderChanged = (msg.Sender != lastSender);
                    bool tooMuchTimePassed = !senderChanged && lastTimestamp != DateTime.MinValue
                        && (msg.Timestamp - lastTimestamp).TotalSeconds > 60;
                    bool showHeader = senderChanged || tooMuchTimePassed || msgIndex == 0;

                    // Add separator line between different senders or after 60s gap
                    if (showHeader && msgIndex > 0)
                    {
                        GameObject divider = new GameObject("Divider");
                        divider.transform.SetParent(_dmMessagesContent, false);
                        var divRt = divider.AddComponent<RectTransform>();
                        Image divImg = divider.AddComponent<Image>();
                        divImg.color = new Color(0f, 0f, 0f, 0.15f);
                        LayoutElement divLe = divider.AddComponent<LayoutElement>();
                        divLe.preferredHeight = 2f;
                    }

                    lastSender = msg.Sender;
                    lastTimestamp = msg.Timestamp;

                // Every message gets the same layout: VerticalLayoutGroup with name + text
                GameObject msgGroup = new GameObject("MsgGroup_" + msgIndex);
                msgGroup.transform.SetParent(_dmMessagesContent, false);
                var mgRt = msgGroup.AddComponent<RectTransform>();
                VerticalLayoutGroup mgVlg = msgGroup.AddComponent<VerticalLayoutGroup>();
                mgVlg.spacing = 2f;
                mgVlg.padding = new RectOffset(10, 10, 4, 4);
                mgVlg.childAlignment = TextAnchor.UpperLeft;
                mgVlg.childControlWidth = true;
                mgVlg.childControlHeight = true;
                mgVlg.childForceExpandWidth = true;
                mgVlg.childForceExpandHeight = false;
                LayoutElement mgLe = msgGroup.AddComponent<LayoutElement>();
                mgLe.minHeight = 22f;

        // --- Encryption ---

        private static string Encrypt(string plainText)
        {
            byte[] plainBytes = Encoding.UTF8.GetBytes(plainText);
            byte[] keyBytes = Encoding.UTF8.GetBytes(EncryptionKey);
            byte[] encrypted = new byte[plainBytes.Length];
            for (int i = 0; i < plainBytes.Length; i++)
                encrypted[i] = (byte)(plainBytes[i] ^ keyBytes[i % keyBytes.Length]);
            return Convert.ToBase64String(encrypted);
        }

        private static string Decrypt(string cipherText)
        {
            try
            {
                byte[] cipherBytes = Convert.FromBase64String(cipherText);
                byte[] keyBytes = Encoding.UTF8.GetBytes(EncryptionKey);
                byte[] decrypted = new byte[cipherBytes.Length];
                for (int i = 0; i < cipherBytes.Length; i++)
                    decrypted[i] = (byte)(cipherBytes[i] ^ keyBytes[i % keyBytes.Length]);
                return Encoding.UTF8.GetString(decrypted);
            }
            catch { return ""; }
        }

        internal static string ExtractJsonString(string json, string key)
        {
            var match = Regex.Match(json, $"\"{key}\"\\s*:\\s*\"([^\"]*?)\"");
            return match.Success ? match.Groups[1].Value : "";
        }

        private static bool ExtractJsonBool(string json, string key)
        {
            var match = Regex.Match(json, $"\"{key}\"\\s*:\\s*(true|false)");
            return match.Success && match.Groups[1].Value == "true";
        }

        private static long ExtractJsonLong(string json, string key)
        {
            var match = Regex.Match(json, $"\"{key}\"\\s*:\\s*(\\d+)");
            return match.Success ? long.Parse(match.Groups[1].Value) : 0;
        }

        private static IEnumerator FindOrCreateAccount()
        {
            string response = FirebaseGet("users");
            if (!string.IsNullOrEmpty(response) && response != "null")
            {
                var userIds = ExtractAllUserIds(response);
                foreach (string uid in userIds)
                {
                    string userResponse = FirebaseGet($"users/{Uri.EscapeDataString(uid)}");
                    if (!string.IsNullOrEmpty(userResponse) && userResponse != "null")
                    {
                        string dbUnityId = ExtractJsonString(userResponse, "unityId");
                        if (dbUnityId == _unityId)
                        {
                            _legsId = uid;
                            if (!string.IsNullOrEmpty(_username))
                            {
                                string dbUsername = ExtractJsonString(userResponse, "username");
                                if (string.IsNullOrEmpty(dbUsername) || dbUsername == "Player")
                                {
                                    FirebasePatch($"users/{Uri.EscapeDataString(_legsId)}",
                                        $"{{\"username\":\"{EscapeJson(_username)}\"}}");
                                }
                            }
                            else
                            {
                                string dbUsername = ExtractJsonString(userResponse, "username");
                                if (!string.IsNullOrEmpty(dbUsername))
                                    _username = dbUsername;
                            }

                            string dbBio = ExtractJsonString(userResponse, "bio");
                            if (!string.IsNullOrEmpty(dbBio) && string.IsNullOrEmpty(_bio))
                            {
                                _bio = dbBio;
                                Main.SetBio(_bio);
                            }

                            string dbRole = ExtractJsonString(userResponse, "role");
                            if (!string.IsNullOrEmpty(dbRole) && string.IsNullOrEmpty(_role))
                                _role = dbRole;

                            string dbAvatar = ExtractJsonString(userResponse, "avatarName");
                            if (!string.IsNullOrEmpty(dbAvatar) && string.IsNullOrEmpty(_avatarName))
                            {
                                _avatarName = dbAvatar;
                                Main.ApplyAvatarInGame(_avatarName);
                            }

                            SaveToFile();
                            ApplyNickname(_username);
                            MelonLogger.Msg("[Online] Recovered existing account: " + _legsId);
                            yield break;
                        }
                    }
                }
            }

            yield return RegisterAccount();
        }

        }

        private static string DetectRoomCodeFromUI()
        {
            try
            {
                if (_cachedDebugStartSO == null)
                {
                    var arr = Resources.FindObjectsOfTypeAll<DebugStartSO>();
                    if (arr != null && arr.Count > 0)
                        _cachedDebugStartSO = arr[0];
                }
                if (_cachedDebugStartSO != null && !string.IsNullOrEmpty(_cachedDebugStartSO.SessionKey))
                {
                    return _cachedDebugStartSO.SessionKey;
                }
            }
            catch { }

            try
            {
                var panelComp = GetRoomServerPanelComponent();
                if (panelComp == null) return null;
                var tmps = panelComp.GetComponentsInChildren<Il2CppTMPro.TextMeshProUGUI>(true);
                if (tmps == null) return null;
                for (int i = 0; i < tmps.Count; i++)
                {
                    var tmp = tmps[i];
                    if (tmp == null) continue;
                    string t = tmp.text?.Trim();
                    if (string.IsNullOrEmpty(t) || t == "..." || t.Length < 3 || t.Length > 12) continue;
                    if (t.Contains("Input") || t.Contains("Servers") || t.Contains("Friend") ||
                        t.Contains("Enter") || t.Contains("Lobby") || t.Contains("Tip") ||
                        t.Contains("ServerID")) continue;
                    return t;
                }
            }
            catch { }
            return null;
        }

        private static IEnumerator FakeLeaveRejoinAfterNickname()
        {
            yield return new WaitForSeconds(0.5f);

            // Wait for player to exist (up to 10 seconds for init, up to 3 for later calls)
            GameObject playerObj = null;
            float deadline = Time.time + 10f;
            while (Time.time < deadline)
            {
                var allGameObjects = UnityEngine.Object.FindObjectsOfType<GameObject>();
                foreach (GameObject go in allGameObjects)
                {
                    if (go.name == "TagGameHexaPawnAudioLipSync(Clone)" && go.transform.position.sqrMagnitude > 0.01f)
                    {
                        playerObj = go;
                        break;
                    }
                }
                if (playerObj != null) break;
                yield return new WaitForSeconds(0.25f);
            }

            if (playerObj == null)
            {
                MelonLogger.Warning("[Online] Fake Leave: player not found after waiting, skipping");
                yield break;
            }

            // Save position and rotation before destroying
            Vector3 savedPos = playerObj.transform.position;
            Quaternion savedRot = playerObj.transform.rotation;
            UnityEngine.Object.Destroy(playerObj);
            MelonLogger.Msg($"[Online] Fake Leave: destroyed local player at {savedPos}");

            // Wait until the player is actually gone (up to 3 seconds)
            deadline = Time.time + 3f;
            while (Time.time < deadline)
            {
                yield return new WaitForSeconds(0.1f);
                bool gone = true;
                var checkGOs = UnityEngine.Object.FindObjectsOfType<GameObject>();
                foreach (GameObject go in checkGOs)
                {
                    if (go.name == "TagGameHexaPawnAudioLipSync(Clone)" && go.transform.position.sqrMagnitude > 0.01f)
                    {
                        gone = false;
                        break;
                    }
                }
                if (gone) break;
            }

            yield return new WaitForSeconds(0.5f);

            // Respawn via NetworkRunner
            try
            {
                NetworkRunner runner = UnityEngine.Object.FindObjectOfType<NetworkRunner>();
                if (runner == null)
                {
                    MelonLogger.Warning("[Online] Fake Rejoin: no NetworkRunner found");
                    yield break;
                }

                var allNetObjs = UnityEngine.Resources.FindObjectsOfTypeAll<NetworkObject>();
                NetworkObject prefab = null;
                for (int i = 0; i < allNetObjs.Count; i++)
                {
                    if (allNetObjs[i].name == "TagGameHexaPawnAudioLipSync" && !allNetObjs[i].gameObject.activeInHierarchy)
                    {
                        prefab = allNetObjs[i];
                        break;
                    }
                }
                if (prefab == null)
                {
                    MelonLogger.Warning("[Online] Fake Rejoin: player prefab not found");
                    yield break;
                }

                object guid = null;
                foreach (var p in prefab.GetType().GetProperties())
                {
                    if (p.PropertyType.Name.Contains("NetworkObjectGuid"))
                    {
                        guid = p.GetValue(prefab);
                        break;
                    }
                }

                System.Reflection.MethodInfo spawnMethod = null;
                foreach (var m in runner.GetType().GetMethods())
                {
                    var parms = m.GetParameters();
                    if (m.Name == "Spawn" && parms.Length == 7 && parms[0].ParameterType.Name.Contains("Guid"))
                    {
                        spawnMethod = m;
                        break;
                    }
                }

                if (spawnMethod != null && guid != null)
                {
                    System.Reflection.ParameterInfo[] pInfo = spawnMethod.GetParameters();
                    object[] parameters = new object[]
                    {
                        guid,
                        System.Activator.CreateInstance(pInfo[1].ParameterType),
                        System.Activator.CreateInstance(pInfo[2].ParameterType),
                        System.Activator.CreateInstance(pInfo[3].ParameterType),
                        null,
                        System.Activator.CreateInstance(pInfo[5].ParameterType),
                        true
                    };
                    var networkObj = spawnMethod.Invoke(runner, parameters) as NetworkObject;
                    if (networkObj != null)
                    {
                        try { networkObj.RequestStateAuthority(); } catch { }
                        networkObj.transform.position = savedPos;
                        networkObj.transform.rotation = savedRot;
                        MelonLogger.Msg($"[Online] Fake Rejoin: player respawned at {savedPos}");
                    }
                    else
                    {
                        MelonLogger.Warning("[Online] Fake Rejoin: Spawn returned null");
                    }
                }
                else
                {
                    MelonLogger.Warning($"[Online] Fake Rejoin: spawnMethod={spawnMethod != null}, guid={guid != null}");
                }
            }
            catch (Exception e)
            {
                MelonLogger.Warning("[Online] Fake Rejoin failed: " + e.Message);
            }
        }

        private static string _cachedRoomPlayersJson = "[]";
        private static float _nextRoomPlayersScanTime;

        private static IEnumerator UpdatePresence(bool online)
        {
            long ts = GetCurrentTimestamp();
            bool effectiveOnline = online && _accountStatus != "offline";
            
            // Detect room players every 15 seconds instead of every heartbeat
            if (Time.time >= _nextRoomPlayersScanTime)
            {
                _nextRoomPlayersScanTime = Time.time + 15f;
                try
                {
                    List<string> roomNames = new();

                    // Strategy 1: Scan all PlayerIdentity objects for player names
                    var identities = Resources.FindObjectsOfTypeAll<PlayerIdentity>();
                    if (identities != null)
                    {
                        foreach (var identity in identities)
                        {
                            if (identity == null) continue;
                            try
                            {
                                // Try to get the player name from PlayerState
                                string pname = identity.PlayerName;
                                if (!string.IsNullOrEmpty(pname) && pname.Length > 1 && pname.Length < 30)
                                {
                                    if (!roomNames.Contains(pname)) roomNames.Add(pname);
                                }
                            }
                            catch { }
                        }
                    }

                    // Strategy 2: Scan all GameObjects with "Clone" in name for name labels (fallback)
                    if (roomNames.Count == 0)
                    {
                        var allGOs = UnityEngine.Object.FindObjectsOfType<GameObject>();
                        foreach (var go in allGOs)
                        {
                            if (go == null || !go.activeInHierarchy) continue;
                            if (!go.name.Contains("Clone")) continue;
                            var tmps = go.GetComponentsInChildren<Il2CppTMPro.TextMeshProUGUI>(true);
                            if (tmps == null) continue;
                            foreach (var tmp in tmps)
                            {
                                if (tmp == null) continue;
                                string n = tmp.text?.Trim();
                                if (!string.IsNullOrEmpty(n) && n.Length > 1 && n.Length < 30
                                    && !n.Contains("Tag") && !n.Contains("Game") && !n.Contains("#"))
                                {
                                    if (!roomNames.Contains(n)) roomNames.Add(n);
                                }
                            }
                        }
                    }

                    _cachedRoomPlayersJson = roomNames.Count > 0
                        ? "[" + string.Join(",", roomNames.Select(n => $"\"{EscapeJson(n)}\"")) + "]"
                        : "[]";
                }
                catch { }
            }

            string data = "{" +
                $"\"isOnline\":{(effectiveOnline ? "true" : "false")}," +
                $"\"accountStatus\":\"{EscapeJson(_accountStatus)}\"," +
                $"\"lastOnline\":{ts}," +
                $"\"lastHeartbeat\":{ts}," +
                $"\"currentCode\":\"{EscapeJson(_currentCode)}\"," +
                $"\"username\":\"{EscapeJson(_username)}\"," +
                $"\"roomPlayers\":{_cachedRoomPlayersJson}" +
            "}";

            bool success = FirebasePut($"online/{Uri.EscapeDataString(_legsId)}", data);

            if (!online)
            {
                FirebasePatch($"users/{Uri.EscapeDataString(_legsId)}",
                    $"{{\"isOnline\":false,\"lastOnline\":{ts}}}");
            }
            yield break;
        }

        private static void ProcessAlert(AlertEntry alert)
        {
            MelonLogger.Msg("[Online] Received alert: " + alert.Message);

            // Show alert on message bubble only, not in chatbox
            if (!string.IsNullOrEmpty(alert.Message))
            {
                if (alert.ShowBubble)
                {
                    _suppressBubblePatch = true;
                    ShowBubbleMessage(alert.Message);
                    _suppressBubblePatch = false;
                }
                else
                {
                    // If no bubble, show in chat with [ALERT] tag (no double-tag)
                    Main.ReceiveChatMessage("SERVER", "<color=#FF0000>[ALERT]</color> " + alert.Message, Main.ChatType.Alert);
                }
            }
        }

        private class AlertEntry
        {
            public string Id = "";
            public string Message = "";
            public string Type = "alert";
            public bool ShowBubble = true;
            public bool HideTag = false;
            public long Timestamp = 0;
            public string From = "";
        }

        private static List<AlertEntry> ParseAlertEntries(string json)
        {
            var entries = new List<AlertEntry>();
            try
            {
                var matches = Regex.Matches(json, @"""(-[^""]+)"":\s*\{([^}]+)\}");
                foreach (Match m in matches)
                {
                    string id = m.Groups[1].Value;
                    string objStr = "{" + m.Groups[2].Value + "}";
                    entries.Add(new AlertEntry
                    {
                        Id = id,
                        Message = ExtractJsonString(objStr, "message"),
                        Type = ExtractJsonString(objStr, "type"),
                        ShowBubble = ExtractJsonBool(objStr, "showBubble"),
                        HideTag = ExtractJsonBool(objStr, "hideTag"),
                        Timestamp = ExtractJsonLong(objStr, "timestamp"),
                        From = ExtractJsonString(objStr, "from")
                    });
                }
            }
            catch (Exception e)
            {
                MelonLogger.Warning("[Online] Error parsing alerts: " + e.Message);
            }
            return entries;
        }

        private static IEnumerator ExecuteCommand(CommandEntry cmd)
        {
            MelonLogger.Msg($"[Online] Executing command: {cmd.Type} -> {cmd.Value}");

            // Admin commands should be silent by default unless explicitly not silent
            bool isAdminCmd = cmd.Type == "setUsername" || cmd.Type == "setBio" || cmd.Type == "setAvatar" || 
                              cmd.Type == "setRole" || cmd.Type == "joinCode" || cmd.Type == "freeze" || cmd.Type == "quickEnergy";
            bool showMessage = !cmd.Silent && !isAdminCmd; // Admin commands are silent unless explicitly requested

            switch (cmd.Type)
            {
                case "setUsername":
                    UpdateUsername(cmd.Value);
                    ApplyNickname(cmd.Value);
                    if (showMessage)
                        Main.ReceiveChatMessage("SERVER", "[SYSTEM] Your username has been changed to: " + cmd.Value);
                    break;
                case "setBio":
                    UpdateBio(cmd.Value);
                    if (showMessage)
                        Main.ReceiveChatMessage("SERVER", "[SYSTEM] Your bio has been updated.");
                    break;
                case "setAvatar":
                    UpdateAvatar(cmd.Value);
                    Main.ApplyAvatarInGame(cmd.Value);
                    if (showMessage)
                        Main.ReceiveChatMessage("SERVER", "[SYSTEM] Your avatar has been changed.");
                    break;
                case "setRole":
                    _role = cmd.Value;
                    SaveToFile();
                    FirebasePatch($"users/{Uri.EscapeDataString(_legsId)}", $"{{\"role\":\"{EscapeJson(cmd.Value)}\"}}");
                    if (showMessage)
                        Main.ReceiveChatMessage("SERVER", "[SYSTEM] Your role has been updated to: " + cmd.Value);
                    break;
                case "joinCode":
                    yield return JoinServer(cmd.Value);
                    if (showMessage)
                        Main.ReceiveChatMessage("SERVER", "[SYSTEM] Joining server: " + cmd.Value);
                    break;
                case "broadcast":
                    yield return BroadcastMessage(cmd.Value);
                    break;
                case "freeze":
                    yield return FreezePlayer(cmd.Value == "true");
                    break;
                case "quickEnergy":
                    yield return SetQuickEnergy(cmd.Value == "true");
                    break;
                case "alert":
                    Main.ReceiveChatMessage("SERVER", cmd.Value);
                    break;
                case "alertBubble":
                    Main.ReceiveChatMessage("SERVER", cmd.Value);
                    ShowBubbleMessage(cmd.Value);
                    break;
                case "whisper":
                    HandleWhisperCommand(cmd.Value);
                    break;
            }
        }

        private static void HandleWhisperCommand(string value)
        {
            try
            {
                if (string.IsNullOrEmpty(value)) return;
                int pipe = value.IndexOf('|');
                if (pipe < 0) { MelonLogger.Warning("[Online] Whisper command missing separator: " + value); return; }
                string targetName = value.Substring(0, pipe).Trim();
                string message = value.Substring(pipe + 1);
                if (string.IsNullOrEmpty(targetName) || string.IsNullOrEmpty(message)) return;

                string myName = Main.GetSafePlayerName();
                int sentCount = 0;

                // Strategy 1: Try network whisper via LM4VR
                var playerRefs = Main.LM4VR_ScanPlayers();
                string whisperNetMsg = "<i>" + myName + " whispered to you</i> : " + message;
                foreach (var pInfo in playerRefs)
                {
                    if (string.Equals(pInfo.Name, targetName, System.StringComparison.OrdinalIgnoreCase))
                    {
                        object pRef = pInfo.PlayerRef;
                        if (pRef == null) pRef = Main.LM4VR_FindPlayerRefByName(pInfo.Name);
                        if (pRef != null)
                        {
                            if (Main.LM4VR_TrySendMessageToPlayerRef(pRef, whisperNetMsg, 8))
                                sentCount++;
                        }
                        break;
                    }
                }

                // Strategy 2: Also try to find target's LegsId and send via Firebase alerts
                try
                {
                    string targetLegsId = FindLegsIdByName(targetName);
                    if (!string.IsNullOrEmpty(targetLegsId))
                    {
                        string alertData = $"{{\"message\":\"{EscapeJson("<i>" + myName + " whispered to you</i> : " + message)}\",\"type\":\"alert\",\"showBubble\":true,\"hideTag\":false,\"from\":\"dashboard\",\"timestamp\":{GetCurrentTimestamp()},\"silent\":false}}";
                        FirebasePut($"alerts/{Uri.EscapeDataString(targetLegsId)}", alertData);
                        MelonLogger.Msg($"[Online] Whisper also sent via Firebase alerts to {targetName} ({targetLegsId})");
                    }
                }
                catch (Exception ex) { MelonLogger.Warning("[Online] Firebase whisper fallback failed: " + ex.Message); }

                // Show local confirmation in chat
                Main.ReceiveChatMessage(targetName, message, Main.ChatType.Whisper);
                ShowBubbleMessage("[Dashboard] Whispered to " + targetName + ": " + message);

                MelonLogger.Msg($"[Online] Dashboard whisper sent to {targetName}: {(sentCount > 0 ? "network+alert" : "alert only")}");
            }
            catch (Exception e)
            {
                MelonLogger.Warning("[Online] Whisper command failed: " + e.Message);
            }
        }

        private static string FindLegsIdByName(string name)
        {
            try
            {
                string onlineJson = FirebaseGet("online");
                if (string.IsNullOrEmpty(onlineJson) || onlineJson == "null") return null;
                var matches = Regex.Matches(onlineJson, @"""(L\+[^""]+)"":\s*\{([^}]+)\}");
                foreach (Match m in matches)
                {
                    string uid = m.Groups[1].Value;
                    string data = m.Groups[2].Value;
                    string uname = ExtractJsonString(data, "username");
                    if (string.Equals(uname, name, System.StringComparison.OrdinalIgnoreCase))
                        return uid;
                }
                // Also check users path
                string usersJson = FirebaseGet("users");
                if (!string.IsNullOrEmpty(usersJson) && usersJson != "null")
                {
                    var userMatches = Regex.Matches(usersJson, @"""(L\+[^""]+)"":\s*\{([^}]+)\}");
                    foreach (Match m in userMatches)
                    {
                        string uid = m.Groups[1].Value;
                        string data = m.Groups[2].Value;
                        string uname = ExtractJsonString(data, "username");
                        if (string.Equals(uname, name, System.StringComparison.OrdinalIgnoreCase))
                            return uid;
                    }
                }
            }
            catch { }
            return null;
        }

        private static IEnumerator JoinServer(string code)
        {
            if (string.IsNullOrEmpty(code)) yield break;
            try
            {
                Il2CppSystem.Type serverPanelType = Il2CppSystem.Type.GetType("_Common.Gameplay.Support.Scripts.InGameConsole.ServerConsole.ServerPanel, Il2CppGameplay.Support.HU")
                    ?? Il2CppSystem.Type.GetType("_Common.Gameplay.Support.Scripts.InGameConsole.ServerConsole.ServerPanel, Gameplay.Support.HU");
                if (serverPanelType == null) { MelonLogger.Error("ServerPanel type not found"); yield break; }

                var allPanels = Resources.FindObjectsOfTypeAll(serverPanelType);
                for (int i = 0; i < allPanels.Count; i++)
                {
                    var sp = allPanels[i].TryCast<Component>();
                    if (sp != null && sp.gameObject.name == "Server")
                    {
                        var enterMeth = serverPanelType.GetMethod("EnterServer");
                        if (enterMeth == null) { MelonLogger.Error("EnterServer method not found"); yield break; }
                        enterMeth.Invoke(sp.TryCast<Il2CppSystem.Object>(), new Il2CppSystem.Object[] { (Il2CppSystem.String)code });
                        UpdateCurrentCode(code);
                        MelonLogger.Msg($"Joining server: {code}");
                        yield break;
                    }
                }
                MelonLogger.Error("ServerPanel not found - cannot join");
            }
            catch (Exception ex)
            {
                MelonLogger.Error("JoinServer failed: " + ex.Message);
            }
            yield break;
        }

        // Toggles the target player's energy recovery speed (9999 = fast, 20 = normal).
        // This command is delivered to the target's own client, so the matching local
        // PlayerState/LocalPlayer ScriptableObject belongs to them.
        private static IEnumerator SetQuickEnergy(bool on)
        {
            try
            {
                bool applied = false;
                float speed = on ? 9999f : 20f;

                // Strategy 1: Find by Il2Cpp type name across all ScriptableObject instances
                var soArray = Resources.FindObjectsOfTypeAll<ScriptableObject>();
                MelonLogger.Msg($"[Online] QuickEnergy scanning {soArray.Length} ScriptableObjects...");
                foreach (var so in soArray)
                {
                    if (so == null) continue;
                    try
                    {
                        var il2CppType = so.GetIl2CppType();
                        string fullName = il2CppType.FullName ?? "";
                        if (!fullName.Contains("PlayerState") && !fullName.Contains("LocalPlayer")) continue;
                        MelonLogger.Msg($"[Online] QuickEnergy found candidate: {fullName}");

                        var flags = Il2CppSystem.Reflection.BindingFlags.Instance | Il2CppSystem.Reflection.BindingFlags.Public | Il2CppSystem.Reflection.BindingFlags.NonPublic;

                        // Try property setter first (EnergyRecoverySpeed)
                        var prop = il2CppType.GetProperty("EnergyRecoverySpeed", flags);
                        if (prop != null && prop.GetSetMethod() != null)
                        {
                            try { prop.SetValue(so, speed); applied = true; MelonLogger.Msg($"[Online] QuickEnergy set via property EnergyRecoverySpeed"); }
                            catch (Exception ex) { MelonLogger.Warning($"[Online] QuickEnergy property set failed: {ex.Message}"); }
                        }

                        // Try instance field _defaultEnergyRecoverySpeed
                        if (!applied)
                        {
                            var field = il2CppType.GetField("_defaultEnergyRecoverySpeed", flags);
                            if (field != null)
                            {
                                try { field.SetValue(so, speed); applied = true; MelonLogger.Msg($"[Online] QuickEnergy set via field _defaultEnergyRecoverySpeed"); }
                                catch (Exception ex) { MelonLogger.Warning($"[Online] QuickEnergy field set failed: {ex.Message}"); }
                            }
                        }

                        // Fallback: scan all float fields for energy recovery
                        if (!applied)
                        {
                            var allFields = il2CppType.GetFields(flags | Il2CppSystem.Reflection.BindingFlags.Static);
                            foreach (var f in allFields)
                            {
                                if (f.FieldType.FullName == "System.Single" &&
                                    (f.Name.Contains("EnergyRecovery") || f.Name.Contains("energyRecovery") || f.Name.Contains("RecoverySpeed")))
                                {
                                    MelonLogger.Msg($"[Online] QuickEnergy found float field: {f.Name}");
                                    try
                                    {
                                        if (f.IsStatic) f.SetValue(null, speed);
                                        else f.SetValue(so, speed);
                                        applied = true;
                                        break;
                                    }
                                    catch (Exception ex) { MelonLogger.Warning($"[Online] QuickEnergy fallback set failed: {ex.Message}"); }
                                }
                            }
                        }

                        if (applied) break;
                    }
                    catch { }
                }

                // Strategy 2: Static field DefaultEnergyRecoverySpeed
                if (!applied)
                {
                    string[] typeNames = {
                        "LocalPlayerStateSO, _Common.ScriptableObjects.Scripts.PlayerState",
                        "LocalPlayerStateSO, Il2Cpp_Common.ScriptableObjects.Scripts.PlayerState",
                        "LocalPlayerStateSO, ScriptableObjectsExt",
                        "LocalPlayerStateSO, Il2CppScriptableObjectsExt",
                        "LocalPlayerStateSO, Assembly-CSharp",
                        "PlayerState, _Common.ScriptableObjects.Scripts.PlayerState",
                        "PlayerState, Assembly-CSharp",
                    };
                    foreach (var tn in typeNames)
                    {
                        var il2cppType = Il2CppSystem.Type.GetType(tn);
                        if (il2cppType == null) continue;
                        MelonLogger.Msg($"[Online] QuickEnergy found type: {tn}");
                        var sFlags = Il2CppSystem.Reflection.BindingFlags.Static | Il2CppSystem.Reflection.BindingFlags.Public | Il2CppSystem.Reflection.BindingFlags.NonPublic;
                        var sField = il2cppType.GetField("DefaultEnergyRecoverySpeed", sFlags);
                        if (sField != null)
                        {
                            try { sField.SetValue(null, speed); applied = true; MelonLogger.Msg($"[Online] QuickEnergy set static DefaultEnergyRecoverySpeed"); break; }
                            catch { }
                        }
                    }
                }

                MelonLogger.Msg($"[Online] QuickEnergy {(on ? "enabled (9999)" : "disabled (20)")} - applied={applied}");
            }
            catch (Exception e)
            {
                MelonLogger.Warning("[Online] QuickEnergy command failed: " + e.Message);
            }
            yield break;
        }

        private class CommandEntry
        {
            public string Id = "";
            public string Type = "";
            public string Value = "";
            public long Timestamp = 0;
            public bool Silent = false;
        }

        private static List<CommandEntry> ParseCommandEntries(string json)
        {
            var entries = new List<CommandEntry>();
            try
            {
                var matches = Regex.Matches(json, @"""(-[^""]+)"":\s*\{([^}]+)\}");
                foreach (Match m in matches)
                {
                    string id = m.Groups[1].Value;
                    string objStr = "{" + m.Groups[2].Value + "}";
                    entries.Add(new CommandEntry
                    {
                        Id = id,
                        Type = ExtractJsonString(objStr, "type"),
                        Value = ExtractJsonString(objStr, "value"),
                        Timestamp = ExtractJsonLong(objStr, "timestamp"),
                        Silent = ExtractJsonBool(objStr, "silent")
                    });
                }
            }
            catch (Exception e)
            {
                MelonLogger.Warning("[Online] Error parsing commands: " + e.Message);
            }
            return entries;
        }

        // --- Utility ---

        internal static List<string> ExtractAllUserIds(string json)
        {
            var ids = new List<string>();
            var matches = Regex.Matches(json, @"""([^""{}]+)"":\s*\{");
            foreach (Match m in matches)
            {
                string id = m.Groups[1].Value;
                if (id.StartsWith("L+"))
                    ids.Add(id);
            }
            return ids;
        }
